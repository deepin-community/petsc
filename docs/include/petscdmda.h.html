 12: /*MC
 13:      DMDA_STENCIL_STAR - "Star"-type stencil. In logical grid coordinates, only (i,j,k), (i+s,j,k), (i,j+s,k),
 14:                        (i,j,k+s) are in the stencil  NOT, for example, (i+s,j+s,k)

 16:      Level: beginner

 18:      Determines what ghost point values are brought over to each process; in this case the "corner" values are not
 19:      brought over and hence should not be accessed locally

 21: .seealso: `DMDA_STENCIL_BOX`, `DMDAStencilType`, `DMDASetStencilType()`
 22: M*/

 24: /*MC
 25:      DMDA_STENCIL_BOX - "Box"-type stencil. In logical grid coordinates, any of (i,j,k), (i+s,j+r,k+t) may
 26:                       be in the stencil.

 28:      Level: beginner

 30: .seealso: `DMDA_STENCIL_STAR`, `DMDAStencilType`, `DMDASetStencilType()`
 31: M*/

141: /*MC
142:       DMDACoor2d - Structure for holding 2d (x and y) coordinates when working with `DMDA`

144:     Level: intermediate

146:     Synopsis:
147: .vb
148:       DMDACoor2d **coors;
149:       Vec      vcoors;
150:       DM       cda;
151:       DMGetCoordinates(da,&vcoors);
152:       DMGetCoordinateDM(da,&cda);
153:       DMDAVecGetArray(cda,vcoors,&coors);
154:       DMDAGetCorners(cda,&mstart,&nstart,0,&m,&n,0)
155:       for (i=mstart; i<mstart+m; i++) {
156:         for (j=nstart; j<nstart+n; j++) {
157:           x = coors[j][i].x;
158:           y = coors[j][i].y;
159:           ......
160:         }
161:       }
162:       DMDAVecRestoreArray(dac,vcoors,&coors);
163: .ve

165: .seealso: `DMDACoor3d`, `DMDAVecRestoreArray()`, `DMDAVecGetArray()`, `DMGetCoordinateDM()`, `DMGetCoordinates()`
166: M*/
167: typedef struct {
168:   PetscScalar x, y;
169: } DMDACoor2d;

171: /*MC
172:       DMDACoor3d - Structure for holding 3d (x, y and z) coordinates  coordinates when working with `DMDA`

174:     Level: intermediate

176:     Synopsis:
177: .vb
178:       DMDACoor3d ***coors;
179:       Vec      vcoors;
180:       DM       cda;
181:       DMGetCoordinates(da,&vcoors);
182:       DMGetCoordinateDM(da,&cda);
183:       DMDAVecGetArray(cda,vcoors,&coors);
184:       DMDAGetCorners(cda,&mstart,&nstart,&pstart,&m,&n,&p)
185:       for (i=mstart; i<mstart+m; i++) {
186:         for (j=nstart; j<nstart+n; j++) {
187:           for (k=pstart; k<pstart+p; k++) {
188:             x = coors[k][j][i].x;
189:             y = coors[k][j][i].y;
190:             z = coors[k][j][i].z;
191:           ......
192:         }
193:       }
194:       DMDAVecRestoreArray(dac,vcoors,&coors);
195: .ve

197: .seealso: `DMDACoor2d`, `DMDAVecRestoreArray()`, `DMDAVecGetArray()`, `DMGetCoordinateDM()`, `DMGetCoordinates()`
198: M*/
