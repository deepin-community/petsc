 37:   PetscInt    as_type;

 39:   PetscBool inv_sig;
 40:   PetscReal tol_quad;        /* tolerance for Dai-Kou dynamic restart */

 44:   PetscBool neg_xi;

 46:   PetscBool diag_scaling;
