  7: /*   VecMergedDot_Private function merges the dot products for gamma, delta and dp */
  8: static PetscErrorCode VecMergedDot_Private(Vec U, Vec W, Vec R, PetscInt normtype, PetscScalar *ru, PetscScalar *wu, PetscScalar *uu)
  9: {
 10:   const PetscScalar *PETSC_RESTRICT PU, *PETSC_RESTRICT PW, *PETSC_RESTRICT PR;
 11:   PetscScalar sumru = 0.0, sumwu = 0.0, sumuu = 0.0;
 12:   PetscInt    j, n;

 14:   VecGetArrayRead(U, (const PetscScalar **)&PU);
 15:   VecGetArrayRead(W, (const PetscScalar **)&PW);
 16:   VecGetArrayRead(R, (const PetscScalar **)&PR);
 17:   VecGetLocalSize(U, &n);

 19:   if (normtype == KSP_NORM_PRECONDITIONED) {
 20:     PetscPragmaSIMD for (j = 0; j < n; j++)
 21:     {
 22:       sumwu += PW[j] * PetscConj(PU[j]);
 23:       sumru += PR[j] * PetscConj(PU[j]);
 24:       sumuu += PU[j] * PetscConj(PU[j]);
 25:     }
 26:   } else if (normtype == KSP_NORM_UNPRECONDITIONED) {
 27:     PetscPragmaSIMD for (j = 0; j < n; j++)
 28:     {
 29:       sumwu += PW[j] * PetscConj(PU[j]);
 30:       sumru += PR[j] * PetscConj(PU[j]);
 31:       sumuu += PR[j] * PetscConj(PR[j]);
 32:     }
 33:   } else if (normtype == KSP_NORM_NATURAL) {
 34:     PetscPragmaSIMD for (j = 0; j < n; j++)
 35:     {
 36:       sumwu += PW[j] * PetscConj(PU[j]);
 37:       sumru += PR[j] * PetscConj(PU[j]);
 38:     }
 39:     sumuu = sumru;
 40:   }

 42:   *ru = sumru;
 43:   *wu = sumwu;
 44:   *uu = sumuu;

 46:   VecRestoreArrayRead(U, (const PetscScalar **)&PU);
 47:   VecRestoreArrayRead(W, (const PetscScalar **)&PW);
 48:   VecRestoreArrayRead(R, (const PetscScalar **)&PR);
 49:   return 0;
 50: }

 53: static PetscErrorCode VecMergedDot2_Private(Vec N, Vec M, Vec W, PetscScalar *wm, PetscScalar *nm)
 54: {

 61:   VecGetArrayRead(M, (const PetscScalar **)&PM);

 65:   {
 66:     sumwm += PW[j] * PetscConj(PM[j]);

 76:   return 0;
 77: }

423:       px[j] = px[j] + alpha1 * pp[j];
424:       pr[j] = pr[j] - alpha1 * ps[j];
425:       pu[j] = pu[j] - alpha1 * pq[j];
426:       pw[j] = pw[j] - alpha1 * pz[j];
427:       pm[j] = pm[j] - alpha1 * pc[j];
428:       pn[j] = pn[j] - alpha1 * pd[j];

430:       lambda[0] += ps[j] * PetscConj(pu[j]);
431:       lambda[1] += pw[j] * PetscConj(pm[j]);
432:       lambda[2] += pw[j] * PetscConj(pq[j]);
433:       lambda[4] += ps[j] * PetscConj(pq[j]);
434:       lambda[6] += pn[j] * PetscConj(pm[j]);
435:       lambda[7] += pn[j] * PetscConj(pq[j]);
436:       lambda[9] += pz[j] * PetscConj(pq[j]);
437:       lambda[10] += pr[j] * PetscConj(pu[j]);
438:       lambda[11] += pw[j] * PetscConj(pu[j]);
439:       lambda[12] += pr[j] * PetscConj(pr[j]);
440:     }
441:     lambda[3]  = PetscConj(lambda[2]);
442:     lambda[5]  = PetscConj(lambda[1]);
443:     lambda[8]  = PetscConj(lambda[7]);
444:     lambda[13] = PetscConj(lambda[11]);
445:     lambda[14] = PetscConj(lambda[0]);
446:   } else if (normtype == KSP_NORM_NATURAL) {
447:     PetscPragmaSIMD for (j = 0; j < n; j++)
448:     {
449:       pa1[j] = (pg1[j] - pg0[j]) / alphaold;
450:       pb1[j] = (ph1[j] - ph0[j]) / alphaold;

452:       pz[j]  = pn[j] + beta0 * pz[j];
453:       pq[j]  = pm[j] + beta0 * pq[j];
454:       ps[j]  = pw[j] + beta0 * ps[j];
455:       pp[j]  = pu[j] + beta0 * pp[j];
456:       pc[j]  = pg0[j] + beta0 * pc[j];
457:       pd[j]  = ph0[j] + beta0 * pd[j];
458:       pa1[j] = pe[j] + beta0 * pa1[j];
459:       pb1[j] = pf[j] + beta0 * pb1[j];

461:       px[j]  = px[j] + alpha0 * pp[j];
462:       pr[j]  = pr[j] - alpha0 * ps[j];
463:       pu[j]  = pu[j] - alpha0 * pq[j];
464:       pw[j]  = pw[j] - alpha0 * pz[j];
465:       pm[j]  = pm[j] - alpha0 * pc[j];
466:       pn[j]  = pn[j] - alpha0 * pd[j];
467:       pg0[j] = pg0[j] - alpha0 * pa1[j];
468:       ph0[j] = ph0[j] - alpha0 * pb1[j];

470:       pg1[j] = pg0[j];
471:       ph1[j] = ph0[j];

473:       pz[j] = pn[j] + beta1 * pz[j];
474:       pq[j] = pm[j] + beta1 * pq[j];
475:       ps[j] = pw[j] + beta1 * ps[j];
476:       pp[j] = pu[j] + beta1 * pp[j];
477:       pc[j] = pg0[j] + beta1 * pc[j];
478:       pd[j] = ph0[j] + beta1 * pd[j];

480:       px[j] = px[j] + alpha1 * pp[j];
481:       pr[j] = pr[j] - alpha1 * ps[j];
482:       pu[j] = pu[j] - alpha1 * pq[j];
483:       pw[j] = pw[j] - alpha1 * pz[j];
484:       pm[j] = pm[j] - alpha1 * pc[j];
485:       pn[j] = pn[j] - alpha1 * pd[j];

487:       lambda[0] += ps[j] * PetscConj(pu[j]);
488:       lambda[1] += pw[j] * PetscConj(pm[j]);
489:       lambda[2] += pw[j] * PetscConj(pq[j]);
490:       lambda[4] += ps[j] * PetscConj(pq[j]);
491:       lambda[6] += pn[j] * PetscConj(pm[j]);
492:       lambda[7] += pn[j] * PetscConj(pq[j]);
493:       lambda[9] += pz[j] * PetscConj(pq[j]);
494:       lambda[10] += pr[j] * PetscConj(pu[j]);
495:       lambda[11] += pw[j] * PetscConj(pu[j]);
496:     }
497:     lambda[3]  = PetscConj(lambda[2]);
498:     lambda[5]  = PetscConj(lambda[1]);
499:     lambda[8]  = PetscConj(lambda[7]);
500:     lambda[13] = PetscConj(lambda[11]);
501:     lambda[14] = PetscConj(lambda[0]);
502:     lambda[12] = lambda[10];
503:   }

505:   VecRestoreArray(vx, (PetscScalar **)&px);
506:   VecRestoreArray(vr, (PetscScalar **)&pr);
507:   VecRestoreArray(vz, (PetscScalar **)&pz);
508:   VecRestoreArray(vw, (PetscScalar **)&pw);
509:   VecRestoreArray(vp, (PetscScalar **)&pp);
510:   VecRestoreArray(vq, (PetscScalar **)&pq);
511:   VecRestoreArray(vc, (PetscScalar **)&pc);
512:   VecRestoreArray(vd, (PetscScalar **)&pd);
513:   VecRestoreArray(vg0, (PetscScalar **)&pg0);
514:   VecRestoreArray(vh0, (PetscScalar **)&ph0);
515:   VecRestoreArray(vg1, (PetscScalar **)&pg1);
516:   VecRestoreArray(vh1, (PetscScalar **)&ph1);
517:   VecRestoreArray(vs, (PetscScalar **)&ps);
518:   VecRestoreArray(va1, (PetscScalar **)&pa1);
519:   VecRestoreArray(vb1, (PetscScalar **)&pb1);
520:   VecRestoreArray(ve, (PetscScalar **)&pe);
521:   VecRestoreArray(vf, (PetscScalar **)&pf);
522:   VecRestoreArray(vm, (PetscScalar **)&pm);
523:   VecRestoreArray(vn, (PetscScalar **)&pn);
524:   VecRestoreArray(vu, (PetscScalar **)&pu);
525:   return 0;
526: }

534: static PetscErrorCode KSPSetUp_PIPECG2(KSP ksp)

537:   KSPSetWorkVecs(ksp, 20);
538:   return 0;

541: /*
542:  KSPSolve_PIPECG2 - This routine actually applies the PIPECG2 method
543: */
544: static PetscErrorCode KSPSolve_PIPECG2(KSP ksp)
545: {
546:   PetscInt    i, n;
547:   PetscScalar alpha[2], beta[2], gamma[2], delta[2], lambda[15];
548:   PetscScalar dps = 0.0, alphaold = 0.0;
549:   PetscReal   dp = 0.0;
550:   Vec         X, B, Z, P, W, Q, U, M, N, R, S, C, D, E, F, G[2], H[2], A1, B1;
551:   Mat         Amat, Pmat;
552:   PetscBool   diagonalscale;
553:   MPI_Comm    pcomm;
554:   MPI_Request req;
555:   MPI_Status  stat;

557:   pcomm = PetscObjectComm((PetscObject)ksp);
558:   PCGetDiagonalScale(ksp->pc, &diagonalscale);

561:   X    = ksp->vec_sol;
562:   B    = ksp->vec_rhs;
563:   M    = ksp->work[0];
564:   Z    = ksp->work[1];
565:   P    = ksp->work[2];
566:   N    = ksp->work[3];
567:   W    = ksp->work[4];
568:   Q    = ksp->work[5];
569:   U    = ksp->work[6];
570:   R    = ksp->work[7];
571:   S    = ksp->work[8];
572:   C    = ksp->work[9];
573:   D    = ksp->work[10];
574:   E    = ksp->work[11];
575:   F    = ksp->work[12];
576:   G[0] = ksp->work[13];
577:   H[0] = ksp->work[14];
578:   G[1] = ksp->work[15];
579:   H[1] = ksp->work[16];
580:   A1   = ksp->work[17];
581:   B1   = ksp->work[18];

583:   PetscMemzero(alpha, 2 * sizeof(PetscScalar));
584:   PetscMemzero(beta, 2 * sizeof(PetscScalar));
585:   PetscMemzero(gamma, 2 * sizeof(PetscScalar));
586:   PetscMemzero(delta, 2 * sizeof(PetscScalar));
587:   PetscMemzero(lambda, 15 * sizeof(PetscScalar));

589:   VecGetLocalSize(B, &n);
590:   PCGetOperators(ksp->pc, &Amat, &Pmat);

592:   ksp->its = 0;
593:   if (!ksp->guess_zero) {
594:     KSP_MatMult(ksp, Amat, X, R); /*  r <- b - Ax  */
595:     VecAYPX(R, -1.0, B);
596:   } else {
597:     VecCopy(B, R); /*  r <- b (x is 0) */
598:   }

600:   KSP_PCApply(ksp, R, U);       /*  u <- Br  */
601:   KSP_MatMult(ksp, Amat, U, W); /*  w <- Au  */

603:   VecMergedDot_Private(U, W, R, ksp->normtype, &gamma[0], &delta[0], &dps); /*  gamma  <- r'*u , delta <- w'*u , dp <- u'*u or r'*r or r'*u depending on ksp_norm_type  */
604:   lambda[10] = gamma[0];
605:   lambda[11] = delta[0];
606:   lambda[12] = dps;

608: #if defined(PETSC_HAVE_MPI_NONBLOCKING_COLLECTIVES)
609:   MPI_Iallreduce(MPI_IN_PLACE, &lambda[10], 3, MPIU_SCALAR, MPIU_SUM, pcomm, &req);
610: #else
611:   MPIU_Allreduce(MPI_IN_PLACE, &lambda[10], 3, MPIU_SCALAR, MPIU_SUM, pcomm);
612:   req = MPI_REQUEST_NULL;
613: #endif

615:   KSP_PCApply(ksp, W, M);       /*  m <- Bw  */
616:   KSP_MatMult(ksp, Amat, M, N); /*  n <- Am  */

618:   KSP_PCApply(ksp, N, G[0]);          /*  g <- Bn  */
619:   KSP_MatMult(ksp, Amat, G[0], H[0]); /*  h <- Ag  */

621:   KSP_PCApply(ksp, H[0], E);    /*  e <- Bh  */
622:   KSP_MatMult(ksp, Amat, E, F); /*  f <- Ae  */

624:   MPI_Wait(&req, &stat);

626:   gamma[0] = lambda[10];
627:   delta[0] = lambda[11];
628:   dp       = PetscSqrtReal(PetscAbsScalar(lambda[12]));

630:   VecMergedDot2_Private(N, M, W, &lambda[1], &lambda[6]); /*  lambda_1 <- w'*m , lambda_4 <- n'*m  */
631:   MPIU_Allreduce(MPI_IN_PLACE, &lambda[1], 1, MPIU_SCALAR, MPIU_SUM, pcomm);
632:   MPIU_Allreduce(MPI_IN_PLACE, &lambda[6], 1, MPIU_SCALAR, MPIU_SUM, pcomm);

634:   lambda[5]  = PetscConj(lambda[1]);
635:   lambda[13] = PetscConj(lambda[11]);

637:   KSPLogResidualHistory(ksp, dp);
638:   KSPMonitor(ksp, 0, dp);
639:   ksp->rnorm = dp;

641:   (*ksp->converged)(ksp, 0, dp, &ksp->reason, ksp->cnvP); /* test for convergence */
642:   if (ksp->reason) return 0;

644:   for (i = 2; i < ksp->max_it; i += 2) {
645:     if (i == 2) {
646:       beta[0]  = 0;
647:       alpha[0] = gamma[0] / delta[0];

649:       gamma[1] = gamma[0] - alpha[0] * lambda[13] - alpha[0] * delta[0] + alpha[0] * alpha[0] * lambda[1];
650:       delta[1] = delta[0] - alpha[0] * lambda[1] - alpha[0] * lambda[5] + alpha[0] * alpha[0] * lambda[6];

652:       beta[1]  = gamma[1] / gamma[0];
653:       alpha[1] = gamma[1] / (delta[1] - beta[1] / alpha[0] * gamma[1]);

655:       VecMergedOpsShort_Private(X, R, Z, W, P, Q, C, D, G[0], H[0], G[1], H[1], S, A1, B1, E, F, M, N, U, ksp->normtype, beta[0], alpha[0], beta[1], alpha[1], lambda);
656:     } else {
657:       beta[0]  = gamma[1] / gamma[0];
658:       alpha[0] = gamma[1] / (delta[1] - beta[0] / alpha[1] * gamma[1]);

660:       gamma[0] = gamma[1];
661:       delta[0] = delta[1];

663:       gamma[1] = gamma[0] - alpha[0] * (lambda[13] + beta[0] * lambda[14]) - alpha[0] * (delta[0] + beta[0] * lambda[0]) + alpha[0] * alpha[0] * (lambda[1] + beta[0] * lambda[2] + beta[0] * lambda[3] + beta[0] * beta[0] * lambda[4]);

665:       delta[1] = delta[0] - alpha[0] * (lambda[1] + beta[0] * lambda[2]) - alpha[0] * (lambda[5] + beta[0] * lambda[3]) + alpha[0] * alpha[0] * (lambda[6] + beta[0] * lambda[7] + beta[0] * lambda[8] + beta[0] * beta[0] * lambda[9]);

667:       beta[1]  = gamma[1] / gamma[0];
668:       alpha[1] = gamma[1] / (delta[1] - beta[1] / alpha[0] * gamma[1]);

670:       VecMergedOps_Private(X, R, Z, W, P, Q, C, D, G[0], H[0], G[1], H[1], S, A1, B1, E, F, M, N, U, ksp->normtype, beta[0], alpha[0], beta[1], alpha[1], lambda, alphaold);
671:     }

673:     gamma[0] = gamma[1];
674:     delta[0] = delta[1];

676: #if defined(PETSC_HAVE_MPI_NONBLOCKING_COLLECTIVES)
677:     MPI_Iallreduce(MPI_IN_PLACE, lambda, 15, MPIU_SCALAR, MPIU_SUM, pcomm, &req);
678: #else
679:     MPIU_Allreduce(MPI_IN_PLACE, lambda, 15, MPIU_SCALAR, MPIU_SUM, pcomm);
680:     req = MPI_REQUEST_NULL;
681: #endif

683:     KSP_PCApply(ksp, N, G[0]);          /*  g <- Bn  */
684:     KSP_MatMult(ksp, Amat, G[0], H[0]); /*  h <- Ag  */

686:     KSP_PCApply(ksp, H[0], E);    /*  e <- Bh  */
687:     KSP_MatMult(ksp, Amat, E, F); /*  f <- Ae */

689:     MPI_Wait(&req, &stat);

691:     gamma[1] = lambda[10];
692:     delta[1] = lambda[11];
693:     dp       = PetscSqrtReal(PetscAbsScalar(lambda[12]));

695:     alphaold = alpha[1];
696:     ksp->its = i;

698:     if (i > 0) {
699:       if (ksp->normtype == KSP_NORM_NONE) dp = 0.0;
700:       ksp->rnorm = dp;
701:       KSPLogResidualHistory(ksp, dp);
702:       KSPMonitor(ksp, i, dp);
703:       (*ksp->converged)(ksp, i, dp, &ksp->reason, ksp->cnvP);
704:       if (ksp->reason) break;
705:     }
706:   }

708:   if (i >= ksp->max_it) ksp->reason = KSP_DIVERGED_ITS;
709:   return 0;
710: }

712: /*MC
713:    KSPPIPECG2 - Pipelined conjugate gradient method with a single non-blocking reduction per two iterations. [](sec_pipelineksp)

715:    Level: intermediate

717:    Notes:
718:    This method has only a single non-blocking reduction per two iterations, compared to 2 blocking for standard CG.  The
719:    non-blocking reduction is overlapped by two matrix-vector products and two preconditioner applications.

721:    MPI configuration may be necessary for reductions to make asynchronous progress, which is important for performance of pipelined methods.
722:    See [](doc_faq_pipelined)

724:    Contributed by:
725:    Manasi Tiwari, Computational and Data Sciences, Indian Institute of Science, Bangalore

727:    Reference:
728:    Manasi Tiwari and Sathish Vadhiyar, "Pipelined Conjugate Gradient Methods for Distributed Memory Systems",
729:    Submitted to International Conference on High Performance Computing, Data and Analytics 2020.

731:    Developer Note:
732:    The implementation code contains a good amount of hand tuned fusion of multiple inner products and similar computations on multiple vectors

734: .seealso: [](chapter_ksp), [](doc_faq_pipelined), [](sec_pipelineksp), `KSPCreate()`, `KSPSetType()`, `KSPCG`, `KSPPIPECG`, `KSPGROPPCG`
735: M*/
736: PETSC_EXTERN PetscErrorCode KSPCreate_PIPECG2(KSP ksp)
737: {
738:   KSPSetSupportedNorm(ksp, KSP_NORM_UNPRECONDITIONED, PC_LEFT, 2);
739:   KSPSetSupportedNorm(ksp, KSP_NORM_PRECONDITIONED, PC_LEFT, 2);
740:   KSPSetSupportedNorm(ksp, KSP_NORM_NATURAL, PC_LEFT, 2);
741:   KSPSetSupportedNorm(ksp, KSP_NORM_NONE, PC_LEFT, 1);

743:   ksp->ops->setup          = KSPSetUp_PIPECG2;
744:   ksp->ops->solve          = KSPSolve_PIPECG2;
745:   ksp->ops->destroy        = KSPDestroyDefault;
746:   ksp->ops->view           = NULL;
747:   ksp->ops->setfromoptions = NULL;
748:   ksp->ops->buildsolution  = KSPBuildSolutionDefault;
749:   ksp->ops->buildresidual  = KSPBuildResidualDefault;
750:   return 0;
751: }
