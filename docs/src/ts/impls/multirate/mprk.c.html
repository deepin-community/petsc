  1: /*
  2:   Code for time stepping with the Multirate Partitioned Runge-Kutta method

  4:   Notes:
  5:   1) The general system is written as
  6:      Udot = F(t,U)
  7:      if one does not split the RHS function, but gives the indexes for both slow and fast components;
  8:   2) The general system is written as
  9:      Usdot = Fs(t,Us,Uf)
 10:      Ufdot = Ff(t,Us,Uf)
 11:      for component-wise partitioned system,
 12:      users should split the RHS function themselves and also provide the indexes for both slow and fast components.
 13:   3) To correct The confusing terminology in the paper, we use 'slow method', 'slow buffer method' and 'fast method' to denote the methods applied to 'slow region', 'slow buffer region' and 'fast region' respectively. The 'slow method' in the original paper actually means the 'slow buffer method'.
 14:   4) Why does the buffer region have to be inside the slow region? The buffer region is treated with a slow method essentially. Applying the slow method to a region with a fast characteristic time scale is apparently not a good choice.

 16:   Reference:
 17:   Emil M. Constantinescu, Adrian Sandu, Multirate Timestepping Methods for Hyperbolic Conservation Laws, Journal of Scientific Computing 2007
 18: */

 69: static PetscErrorCode TSMPRKGenerateTableau2(PetscInt ratio, PetscInt s, const PetscReal Abase[], const PetscReal bbase[], PetscReal A1[], PetscReal b1[], PetscReal A2[], PetscReal b2[])
 70: {
 71:   PetscInt i, j, k, l;

 73:   for (k = 0; k < ratio; k++) {
 74:     /* diagonal blocks */
 75:     for (i = 0; i < s; i++)
 76:       for (j = 0; j < s; j++) {
 77:         A1[(k * s + i) * ratio * s + k * s + j] = Abase[i * s + j];
 78:         A2[(k * s + i) * ratio * s + k * s + j] = Abase[i * s + j] / ratio;
 79:       }
 80:     /* off diagonal blocks */
 81:     for (l = 0; l < k; l++)
 82:       for (i = 0; i < s; i++)
 83:         for (j = 0; j < s; j++) A2[(k * s + i) * ratio * s + l * s + j] = bbase[j] / ratio;
 84:     for (j = 0; j < s; j++) {
 85:       b1[k * s + j] = bbase[j] / ratio;
 86:       b2[k * s + j] = bbase[j] / ratio;
 87:     }
 88:   }
 89:   return 0;
 90: }

 92: static PetscErrorCode TSMPRKGenerateTableau3(PetscInt ratio, PetscInt s, const PetscReal Abase[], const PetscReal bbase[], PetscReal A1[], PetscReal b1[], PetscReal A2[], PetscReal b2[], PetscReal A3[], PetscReal b3[])
 93: {
 94:   PetscInt i, j, k, l, m, n;

 96:   for (k = 0; k < ratio; k++) { /* diagonal blocks of size ratio*s by ratio*s */
 97:     for (l = 0; l < ratio; l++) /* diagonal sub-blocks of size s by s */
 98:       for (i = 0; i < s; i++)
 99:         for (j = 0; j < s; j++) {
100:           A1[((k * ratio + l) * s + i) * ratio * ratio * s + (k * ratio + l) * s + j] = Abase[i * s + j];
101:           A2[((k * ratio + l) * s + i) * ratio * ratio * s + (k * ratio + l) * s + j] = Abase[i * s + j] / ratio;
102:           A3[((k * ratio + l) * s + i) * ratio * ratio * s + (k * ratio + l) * s + j] = Abase[i * s + j] / ratio / ratio;
103:         }
104:     for (l = 0; l < k; l++) /* off-diagonal blocks of size ratio*s by ratio*s */
105:       for (m = 0; m < ratio; m++)
106:         for (n = 0; n < ratio; n++)
107:           for (i = 0; i < s; i++)
108:             for (j = 0; j < s; j++) {
109:               A2[((k * ratio + m) * s + i) * ratio * ratio * s + (l * ratio + n) * s + j] = bbase[j] / ratio / ratio;
110:               A3[((k * ratio + m) * s + i) * ratio * ratio * s + (l * ratio + n) * s + j] = bbase[j] / ratio / ratio;
111:             }
112:     for (m = 0; m < ratio; m++)
113:       for (n = 0; n < m; n++) /* off-diagonal sub-blocks of size s by s in the diagonal blocks */
114:         for (i = 0; i < s; i++)
115:           for (j = 0; j < s; j++) A3[((k * ratio + m) * s + i) * ratio * ratio * s + (k * ratio + n) * s + j] = bbase[j] / ratio / ratio;
116:     for (n = 0; n < ratio; n++)
117:       for (j = 0; j < s; j++) {
118:         b1[(k * ratio + n) * s + j] = bbase[j] / ratio / ratio;
119:         b2[(k * ratio + n) * s + j] = bbase[j] / ratio / ratio;
120:         b3[(k * ratio + n) * s + j] = bbase[j] / ratio / ratio;
121:       }
122:   }
123:   return 0;
124: }

428: /*@C
429:    TSMPRKRegister - register a `TSMPRK` scheme by providing the entries in the Butcher tableau

431:    Not Collective, but the same schemes should be registered on all processes on which they will be used

433:    Input Parameters:
434: +  name - identifier for method
435: .  order - approximation order of method
436: .  s  - number of stages in the base methods
437: .  ratio1 - stepsize ratio at 1st level (e.g. slow/medium)
438: .  ratio2 - stepsize ratio at 2nd level (e.g. medium/fast)
439: .  Af - stage coefficients for fast components(dimension s*s, row-major)
440: .  bf - step completion table for fast components(dimension s)
441: .  cf - abscissa for fast components(dimension s)
442: .  As - stage coefficients for slow components(dimension s*s, row-major)
443: .  bs - step completion table for slow components(dimension s)
444: -  cs - abscissa for slow components(dimension s)

446:    Level: advanced

448:    Note:
449:    Several `TSMPRK` methods are provided, this function is only needed to create new methods.

451: .seealso: [](chapter_ts), `TSMPRK`
452: @*/
453: PetscErrorCode TSMPRKRegister(TSMPRKType name, PetscInt order, PetscInt sbase, PetscInt ratio1, PetscInt ratio2, const PetscReal Asb[], const PetscReal bsb[], const PetscReal csb[], const PetscInt rsb[], const PetscReal Amb[], const PetscReal bmb[], const PetscReal cmb[], const PetscInt rmb[], const PetscReal Af[], const PetscReal bf[], const PetscReal cf[])
454: {
455:   MPRKTableauLink link;
456:   MPRKTableau     t;
457:   PetscInt        s, i, j;


472:   PetscNew(&link);
473:   t = &link->tab;

475:   PetscStrallocpy(name, &t->name);
476:   s        = sbase * ratio1 * ratio2; /*  this is the dimension of the matrices below */
477:   t->order = order;
478:   t->sbase = sbase;
479:   t->s     = s;
480:   t->np    = 2;

482:   PetscMalloc3(s * s, &t->Af, s, &t->bf, s, &t->cf);
483:   PetscArraycpy(t->Af, Af, s * s);
484:   if (bf) {
485:     PetscArraycpy(t->bf, bf, s);
486:   } else
487:     for (i = 0; i < s; i++) t->bf[i] = Af[(s - 1) * s + i];
488:   if (cf) {
489:     PetscArraycpy(t->cf, cf, s);
490:   } else {
491:     for (i = 0; i < s; i++)
492:       for (j = 0, t->cf[i] = 0; j < s; j++) t->cf[i] += Af[i * s + j];
493:   }

495:   if (Amb) {
496:     t->np = 3;
497:     PetscMalloc3(s * s, &t->Amb, s, &t->bmb, s, &t->cmb);
498:     PetscArraycpy(t->Amb, Amb, s * s);
499:     if (bmb) {
500:       PetscArraycpy(t->bmb, bmb, s);
501:     } else {
502:       for (i = 0; i < s; i++) t->bmb[i] = Amb[(s - 1) * s + i];
503:     }
504:     if (cmb) {
505:       PetscArraycpy(t->cmb, cmb, s);
506:     } else {
507:       for (i = 0; i < s; i++)
508:         for (j = 0, t->cmb[i] = 0; j < s; j++) t->cmb[i] += Amb[i * s + j];
509:     }
510:     if (rmb) {
511:       PetscMalloc1(s, &t->rmb);
512:       PetscArraycpy(t->rmb, rmb, s);
513:     } else {
514:       PetscCalloc1(s, &t->rmb);
515:     }
516:   }

518:   PetscMalloc3(s * s, &t->Asb, s, &t->bsb, s, &t->csb);
519:   PetscArraycpy(t->Asb, Asb, s * s);
520:   if (bsb) {
521:     PetscArraycpy(t->bsb, bsb, s);
522:   } else
523:     for (i = 0; i < s; i++) t->bsb[i] = Asb[(s - 1) * s + i];
524:   if (csb) {
525:     PetscArraycpy(t->csb, csb, s);
526:   } else {
527:     for (i = 0; i < s; i++)
528:       for (j = 0, t->csb[i] = 0; j < s; j++) t->csb[i] += Asb[i * s + j];
529:   }
530:   if (rsb) {
531:     PetscMalloc1(s, &t->rsb);
532:     PetscArraycpy(t->rsb, rsb, s);
533:   } else {
534:     PetscCalloc1(s, &t->rsb);
535:   }
536:   link->next      = MPRKTableauList;
537:   MPRKTableauList = link;
538:   return 0;
539: }

627: static PetscErrorCode TSStep_MPRK(TS ts)
628: {
629:   TS_MPRK         *mprk = (TS_MPRK *)ts->data;
630:   Vec             *Y = mprk->Y, *YdotRHS = mprk->YdotRHS, *YdotRHS_fast = mprk->YdotRHS_fast, *YdotRHS_slow = mprk->YdotRHS_slow, *YdotRHS_slowbuffer = mprk->YdotRHS_slowbuffer;
631:   Vec              Yslow, Yslowbuffer, Yfast;
632:   MPRKTableau      tab = mprk->tableau;
633:   const PetscInt   s   = tab->s;
634:   const PetscReal *Af = tab->Af, *cf = tab->cf, *Asb = tab->Asb, *csb = tab->csb;
635:   PetscScalar     *wf = mprk->work_fast, *wsb = mprk->work_slowbuffer;
636:   PetscInt         i, j;
637:   PetscReal        next_time_step = ts->time_step, t = ts->ptime, h = ts->time_step;

639:   for (i = 0; i < s; i++) {
640:     mprk->stage_time = t + h * cf[i];
641:     TSPreStage(ts, mprk->stage_time);
642:     VecCopy(ts->vec_sol, Y[i]);

644:     /* slow buffer region */
645:     for (j = 0; j < i; j++) wsb[j] = h * Asb[i * s + j];
646:     for (j = 0; j < i; j++) VecGetSubVector(YdotRHS[j], mprk->is_slowbuffer, &YdotRHS_slowbuffer[j]);
647:     VecGetSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
648:     VecMAXPY(Yslowbuffer, i, wsb, mprk->YdotRHS_slowbuffer);
649:     VecRestoreSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
650:     for (j = 0; j < i; j++) VecRestoreSubVector(YdotRHS[j], mprk->is_slowbuffer, &YdotRHS_slowbuffer[j]);
651:     /* slow region */
652:     if (mprk->is_slow) {
653:       for (j = 0; j < i; j++) VecGetSubVector(YdotRHS[j], mprk->is_slow, &YdotRHS_slow[j]);
654:       VecGetSubVector(Y[i], mprk->is_slow, &Yslow);
655:       VecMAXPY(Yslow, i, wsb, mprk->YdotRHS_slow);
656:       VecRestoreSubVector(Y[i], mprk->is_slow, &Yslow);
657:       for (j = 0; j < i; j++) VecRestoreSubVector(YdotRHS[j], mprk->is_slow, &YdotRHS_slow[j]);
658:     }

660:     /* fast region */
661:     for (j = 0; j < i; j++) wf[j] = h * Af[i * s + j];
662:     for (j = 0; j < i; j++) VecGetSubVector(YdotRHS[j], mprk->is_fast, &YdotRHS_fast[j]);
663:     VecGetSubVector(Y[i], mprk->is_fast, &Yfast);
664:     VecMAXPY(Yfast, i, wf, mprk->YdotRHS_fast);
665:     VecRestoreSubVector(Y[i], mprk->is_fast, &Yfast);
666:     for (j = 0; j < i; j++) VecRestoreSubVector(YdotRHS[j], mprk->is_fast, &YdotRHS_fast[j]);
667:     if (tab->np == 3) {
668:       Vec         *YdotRHS_medium = mprk->YdotRHS_medium, *YdotRHS_mediumbuffer = mprk->YdotRHS_mediumbuffer;
669:       Vec          Ymedium, Ymediumbuffer;
670:       PetscScalar *wmb = mprk->work_mediumbuffer;

672:       for (j = 0; j < i; j++) wmb[j] = h * tab->Amb[i * s + j];
673:       /* medium region */
674:       if (mprk->is_medium) {
675:         for (j = 0; j < i; j++) VecGetSubVector(YdotRHS[j], mprk->is_medium, &YdotRHS_medium[j]);
676:         VecGetSubVector(Y[i], mprk->is_medium, &Ymedium);
677:         VecMAXPY(Ymedium, i, wmb, mprk->YdotRHS_medium);
678:         VecRestoreSubVector(Y[i], mprk->is_medium, &Ymedium);
679:         for (j = 0; j < i; j++) VecRestoreSubVector(YdotRHS[j], mprk->is_medium, &YdotRHS_medium[j]);
680:       }
681:       /* medium buffer region */
682:       for (j = 0; j < i; j++) VecGetSubVector(YdotRHS[j], mprk->is_mediumbuffer, &YdotRHS_mediumbuffer[j]);
683:       VecGetSubVector(Y[i], mprk->is_mediumbuffer, &Ymediumbuffer);
684:       VecMAXPY(Ymediumbuffer, i, wmb, mprk->YdotRHS_mediumbuffer);
685:       VecRestoreSubVector(Y[i], mprk->is_mediumbuffer, &Ymediumbuffer);
686:       for (j = 0; j < i; j++) VecRestoreSubVector(YdotRHS[j], mprk->is_mediumbuffer, &YdotRHS_mediumbuffer[j]);
687:     }
688:     TSPostStage(ts, mprk->stage_time, i, Y);
689:     /* compute the stage RHS by fast and slow tableau respectively */
690:     TSComputeRHSFunction(ts, t + h * csb[i], Y[i], YdotRHS[i]);
691:   }
692:   TSEvaluateStep(ts, tab->order, ts->vec_sol, NULL);

695:   return 0;
696: }

772:   TS_MPRK         *mprk = (TS_MPRK *)ts->data;
773:   MPRKTableau      tab  = mprk->tableau;
774:   Vec             *Y = mprk->Y, *YdotRHS_fast = mprk->YdotRHS_fast, *YdotRHS_slow = mprk->YdotRHS_slow, *YdotRHS_slowbuffer = mprk->YdotRHS_slowbuffer;
775:   Vec              Yslow, Yslowbuffer, Yfast; /* subvectors for slow and fast components in Y[i] respectively */
776:   PetscInt         s  = tab->s;
777:   const PetscReal *Af = tab->Af, *cf = tab->cf, *Asb = tab->Asb, *csb = tab->csb;
778:   PetscScalar     *wf = mprk->work_fast, *ws = mprk->work_slow, *wsb = mprk->work_slowbuffer;
779:   PetscInt         i, j, computedstages;
780:   PetscReal        next_time_step = ts->time_step, t = ts->ptime, h = ts->time_step;

782:   for (i = 0; i < s; i++) {
783:     mprk->stage_time = t + h * cf[i];
784:     TSPreStage(ts, mprk->stage_time);
785:     /* calculate the stage value for fast and slow components respectively */
786:     VecCopy(ts->vec_sol, Y[i]);
787:     for (j = 0; j < i; j++) wsb[j] = h * Asb[i * s + j];

789:     /* slow buffer region */
790:     if (tab->np == 3 && mprk->is_medium) {
791:       if (tab->rmb[i]) {
792:         VecGetSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
793:         VecISCopy(Y[tab->rmb[i] - 1], mprk->is_slowbuffer, SCATTER_REVERSE, Yslowbuffer);
794:         VecRestoreSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
795:       } else {
796:         PetscScalar *wm = mprk->work_medium;
797:         computedstages  = 0;
798:         for (j = 0; j < i; j++) {
799:           if (tab->rmb[j]) wm[computedstages - tab->sbase + (tab->rmb[j] - 1) % tab->sbase] += wsb[j];
800:           else wm[computedstages++] = wsb[j];
801:         }
802:         VecGetSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
803:         VecMAXPY(Yslowbuffer, computedstages, wm, YdotRHS_slowbuffer);
804:         VecRestoreSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
805:       }
806:     } else {
807:       VecGetSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
808:       VecMAXPY(Yslowbuffer, i, wsb, YdotRHS_slowbuffer);
809:       VecRestoreSubVector(Y[i], mprk->is_slowbuffer, &Yslowbuffer);
810:     }

812:     /* slow region */
813:     if (mprk->is_slow) {
814:       if (tab->rsb[i]) { /* repeat previous stage */
815:         VecGetSubVector(Y[i], mprk->is_slow, &Yslow);
816:         VecISCopy(Y[tab->rsb[i] - 1], mprk->is_slow, SCATTER_REVERSE, Yslow);
817:         VecRestoreSubVector(Y[i], mprk->is_slow, &Yslow);
818:       } else {
819:         computedstages = 0;
820:         for (j = 0; j < i; j++) {
821:           if (tab->rsb[j]) ws[tab->rsb[j] - 1] += wsb[j];
822:           else ws[computedstages++] = wsb[j];
823:         }
824:         VecGetSubVector(Y[i], mprk->is_slow, &Yslow);
825:         VecMAXPY(Yslow, computedstages, ws, YdotRHS_slow);
826:         VecRestoreSubVector(Y[i], mprk->is_slow, &Yslow);
827:         /* only depends on the slow buffer region */
828:         TSComputeRHSFunction(mprk->subts_slow, t + h * csb[i], Y[i], YdotRHS_slow[computedstages]);
829:       }
830:     }

832:     /* fast region */
833:     for (j = 0; j < i; j++) wf[j] = h * Af[i * s + j];
834:     VecGetSubVector(Y[i], mprk->is_fast, &Yfast);
835:     VecMAXPY(Yfast, i, wf, YdotRHS_fast);
836:     VecRestoreSubVector(Y[i], mprk->is_fast, &Yfast);

838:     if (tab->np == 3) {
839:       Vec             *YdotRHS_medium = mprk->YdotRHS_medium, *YdotRHS_mediumbuffer = mprk->YdotRHS_mediumbuffer;
840:       Vec              Ymedium, Ymediumbuffer;
841:       const PetscReal *Amb = tab->Amb, *cmb = tab->cmb;
842:       PetscScalar     *wm = mprk->work_medium, *wmb = mprk->work_mediumbuffer;

844:       for (j = 0; j < i; j++) wmb[j] = h * Amb[i * s + j];
845:       /* medium buffer region */
846:       VecGetSubVector(Y[i], mprk->is_mediumbuffer, &Ymediumbuffer);
847:       VecMAXPY(Ymediumbuffer, i, wmb, YdotRHS_mediumbuffer);
848:       VecRestoreSubVector(Y[i], mprk->is_mediumbuffer, &Ymediumbuffer);
849:       /* medium region */
850:       if (mprk->is_medium) {
851:         if (tab->rmb[i]) { /* repeat previous stage */
852:           VecGetSubVector(Y[i], mprk->is_medium, &Ymedium);
853:           VecISCopy(Y[tab->rmb[i] - 1], mprk->is_medium, SCATTER_REVERSE, Ymedium);
854:           VecRestoreSubVector(Y[i], mprk->is_medium, &Ymedium);
855:         } else {
856:           computedstages = 0;
857:           for (j = 0; j < i; j++) {
858:             if (tab->rmb[j]) wm[computedstages - tab->sbase + (tab->rmb[j] - 1) % tab->sbase] += wmb[j];
859:             else wm[computedstages++] = wmb[j];
860:           }
861:           VecGetSubVector(Y[i], mprk->is_medium, &Ymedium);
862:           VecMAXPY(Ymedium, computedstages, wm, YdotRHS_medium);
863:           VecRestoreSubVector(Y[i], mprk->is_medium, &Ymedium);
864:           /* only depends on the medium buffer region */
865:           TSComputeRHSFunction(mprk->subts_medium, t + h * cmb[i], Y[i], YdotRHS_medium[computedstages]);
866:           /* must be computed after all regions are updated in Y */
867:           TSComputeRHSFunction(mprk->subts_slowbuffer, t + h * csb[i], Y[i], YdotRHS_slowbuffer[computedstages]);
868:         }
869:       }
870:       /* must be computed after fast region and slow region are updated in Y */
871:       TSComputeRHSFunction(mprk->subts_mediumbuffer, t + h * cmb[i], Y[i], YdotRHS_mediumbuffer[i]);
872:     }
873:     if (!(tab->np == 3 && mprk->is_medium)) TSComputeRHSFunction(mprk->subts_slowbuffer, t + h * csb[i], Y[i], YdotRHS_slowbuffer[i]);
874:     TSComputeRHSFunction(mprk->subts_fast, t + h * cf[i], Y[i], YdotRHS_fast[i]);
875:   }

877:   TSEvaluateStep(ts, tab->order, ts->vec_sol, NULL);
878:   ts->ptime += ts->time_step;
879:   ts->time_step = next_time_step;
880:   return 0;
881: }

990: static PetscErrorCode TSSetUp_MPRK(TS ts)
991: {
992:   TS_MPRK    *mprk = (TS_MPRK *)ts->data;
993:   MPRKTableau tab  = mprk->tableau;
994:   DM          dm;

996:   TSRHSSplitGetIS(ts, "slow", &mprk->is_slow);
997:   TSRHSSplitGetIS(ts, "fast", &mprk->is_fast);

1000:   if (tab->np == 3) {
1001:     TSRHSSplitGetIS(ts, "medium", &mprk->is_medium);
1003:     TSRHSSplitGetIS(ts, "mediumbuffer", &mprk->is_mediumbuffer);
1004:     if (!mprk->is_mediumbuffer) { /* let medium buffer cover whole medium region */
1005:       mprk->is_mediumbuffer = mprk->is_medium;
1006:       mprk->is_medium       = NULL;
1007:     }
1008:   }

1010:   /* If users do not provide buffer region settings, the solver will do them automatically, but with a performance penalty */
1011:   TSRHSSplitGetIS(ts, "slowbuffer", &mprk->is_slowbuffer);
1012:   if (!mprk->is_slowbuffer) { /* let slow buffer cover whole slow region */
1013:     mprk->is_slowbuffer = mprk->is_slow;
1014:     mprk->is_slow       = NULL;
1015:   }
1016:   TSCheckImplicitTerm(ts);
1017:   TSMPRKTableauSetUp(ts);
1018:   TSGetDM(ts, &dm);
1019:   DMCoarsenHookAdd(dm, DMCoarsenHook_TSMPRK, DMRestrictHook_TSMPRK, ts);
1020:   DMSubDomainHookAdd(dm, DMSubDomainHook_TSMPRK, DMSubDomainRestrictHook_TSMPRK, ts);
1021:   if (ts->use_splitrhsfunction) {
1022:     ts->ops->step         = TSStep_MPRKSPLIT;
1023:     ts->ops->evaluatestep = TSEvaluateStep_MPRKSPLIT;
1024:     TSMPRKSetSplits(ts);
1025:   } else {
1026:     ts->ops->step         = TSStep_MPRK;
1027:     ts->ops->evaluatestep = TSEvaluateStep_MPRK;
1028:   }
1029:   return 0;
1030: }
