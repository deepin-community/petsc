 16: template <class MemorySpace>
 17: PetscErrorCode VecGetKokkosView_Private(Vec v, PetscScalarKokkosViewType<MemorySpace> *kv, PetscBool overwrite)
 18: {
 19:   Vec_Kokkos *veckok = static_cast<Vec_Kokkos *>(v->spptr);

 21:   VecErrorIfNotKokkos(v);
 22:   if (!overwrite) veckok->v_dual.sync<MemorySpace>(); /* If overwrite=true, no need to sync the space, since caller will overwrite the data */
 23:   *kv = veckok->v_dual.view<MemorySpace>();
 24:   return 0;
 25: }

 27: template <class MemorySpace>
 28: PetscErrorCode VecRestoreKokkosView_Private(Vec v, PetscScalarKokkosViewType<MemorySpace> *kv, PetscBool overwrite)
 29: {
 30:   Vec_Kokkos *veckok = static_cast<Vec_Kokkos *>(v->spptr);

 32:   VecErrorIfNotKokkos(v);
 33:   if (overwrite) veckok->v_dual.clear_sync_state(); /* If overwrite=true, clear the old sync state since user forced an overwrite */
 34:   veckok->v_dual.modify<MemorySpace>();
 35:   PetscObjectStateIncrease((PetscObject)v);
 36:   return 0;
 37: }

 39: template <class MemorySpace>
 40: PetscErrorCode VecGetKokkosView(Vec v, ConstPetscScalarKokkosViewType<MemorySpace> *kv)
 41: {
 42:   Vec_Kokkos *veckok = static_cast<Vec_Kokkos *>(v->spptr);
 43:   VecErrorIfNotKokkos(v);
 44:   veckok->v_dual.sync<MemorySpace>(); /* Sync the space for caller to read */
 45:   *kv = veckok->v_dual.view<MemorySpace>();
 46:   return 0;
 47: }

136: PetscErrorCode VecMin_SeqKokkos(Vec xin, PetscInt *p, PetscReal *val)
137: {
138:   typedef Kokkos::MinLoc<PetscReal, PetscInt>::value_type MinLocValue_t;
139:   ConstPetscScalarKokkosView                              xv;
140:   MinLocValue_t                                           minloc;

142:   PetscLogGpuTimeBegin();
143:   VecGetKokkosView(xin, &xv);
144:   Kokkos::parallel_reduce(
145:     "VecMin", xin->map->n,
146:     KOKKOS_LAMBDA(PetscInt i, MinLocValue_t & lminloc) {
147:       if (PetscRealPart(xv(i)) < lminloc.val) {
148:         lminloc.val = PetscRealPart(xv(i));
149:         lminloc.loc = i;
150:       }
151:     },
152:     Kokkos::MinLoc<PetscReal, PetscInt>(minloc)); /* Kokkos will set minloc properly even if xin is zero-lengthed */
153:   if (p) *p = minloc.loc;
154:   *val = minloc.val;
155:   VecRestoreKokkosView(xin, &xv);
156:   PetscLogGpuTimeEnd();
157:   return 0;
158: }

209: /* y = alpha x + y */
210: PetscErrorCode VecAXPY_SeqKokkos(Vec yin, PetscScalar alpha, Vec xin)
211: {
212:   PetscBool                  xiskok, yiskok;
213:   PetscScalarKokkosView      yv;
214:   ConstPetscScalarKokkosView xv;

216:   if (alpha == (PetscScalar)0.0) return 0;
217:   if (yin == xin) {
218:     VecScale_SeqKokkos(yin, alpha + 1);
219:     return 0;
220:   }
221:   PetscObjectTypeCompareAny((PetscObject)xin, &xiskok, VECSEQKOKKOS, VECMPIKOKKOS, "");
222:   PetscObjectTypeCompareAny((PetscObject)yin, &yiskok, VECSEQKOKKOS, VECMPIKOKKOS, "");
223:   if (xiskok && yiskok) {
224:     PetscLogGpuTimeBegin();
225:     VecGetKokkosView(xin, &xv);
226:     VecGetKokkosView(yin, &yv);
227:     KokkosBlas::axpy(alpha, xv, yv);
228:     VecRestoreKokkosView(xin, &xv);
229:     VecRestoreKokkosView(yin, &yv);
230:     PetscLogGpuTimeEnd();
231:     PetscLogGpuFlops(2.0 * yin->map->n);
232:   } else {
233:     VecAXPY_Seq(yin, alpha, xin);
234:   }
235:   return 0;
236: }

266: struct MDotFunctor {
267:   /* Note the C++ notation for an array typedef */
268:   typedef PetscScalar                           value_type[];
269:   typedef ConstPetscScalarKokkosView::size_type size_type;

271:   /* Tell Kokkos the result array's number of entries. This must be a public value in the functor */
272:   const size_type            value_count;
273:   ConstPetscScalarKokkosView xv, yv[8];

275:   MDotFunctor(ConstPetscScalarKokkosView &xv, const PetscInt ny, /* Number of valid entries in yv[8]. 1 <= ny <= 8 */
276:               ConstPetscScalarKokkosView &yv0, ConstPetscScalarKokkosView &yv1, ConstPetscScalarKokkosView &yv2, ConstPetscScalarKokkosView &yv3, ConstPetscScalarKokkosView &yv4, ConstPetscScalarKokkosView &yv5, ConstPetscScalarKokkosView &yv6, ConstPetscScalarKokkosView &yv7) :
277:     value_count(ny), xv(xv)
278:   {
279:     yv[0] = yv0;
280:     yv[1] = yv1;
281:     yv[2] = yv2;
282:     yv[3] = yv3;
283:     yv[4] = yv4;
284:     yv[5] = yv5;
285:     yv[6] = yv6;
286:     yv[7] = yv7;
287:   }

289:   KOKKOS_INLINE_FUNCTION void operator()(TransposeDotTag, const size_type i, value_type sum) const
290:   {
291:     PetscScalar xval = xv(i);
292:     for (size_type j = 0; j < value_count; ++j) sum[j] += yv[j](i) * xval;
293:   }

295:   KOKKOS_INLINE_FUNCTION void operator()(ConjugateDotTag, const size_type i, value_type sum) const
296:   {
297:     PetscScalar xval = xv(i);
298:     for (size_type j = 0; j < value_count; ++j) sum[j] += PetscConj(yv[j](i)) * xval;
299:   }

301:   // Per https://kokkos.github.io/kokkos-core-wiki/API/core/parallel-dispatch/parallel_reduce.html#requirements
302:   // "when specifying a tag in the policy, the functor's potential init/join/final member functions must also be tagged"
303:   // So we have this kind of duplicated code.
304:   KOKKOS_INLINE_FUNCTION void join(TransposeDotTag, value_type dst, const value_type src) const { join(dst, src); }
305:   KOKKOS_INLINE_FUNCTION void join(ConjugateDotTag, value_type dst, const value_type src) const { join(dst, src); }

307:   KOKKOS_INLINE_FUNCTION void init(TransposeDotTag, value_type sum) const { init(sum); }
308:   KOKKOS_INLINE_FUNCTION void init(ConjugateDotTag, value_type sum) const { init(sum); }

310:   KOKKOS_INLINE_FUNCTION void join(value_type dst, const value_type src) const
311:   {
312:     for (size_type j = 0; j < value_count; j++) dst[j] += src[j];
313:   }

315:   KOKKOS_INLINE_FUNCTION void init(value_type sum) const
316:   {
317:     for (size_type j = 0; j < value_count; j++) sum[j] = 0.0;
318:   }
319: };

321: template <class WorkTag>
322: PetscErrorCode VecMultiDot_Private(Vec xin, PetscInt nv, const Vec yin[], PetscScalar *z)
323: {
324:   PetscInt                   i, j, cur = 0, ngroup = nv / 8, rem = nv % 8, N = xin->map->n;
325:   ConstPetscScalarKokkosView xv, yv[8];
326:   PetscScalarKokkosViewHost  zv(z, nv);

328:   VecGetKokkosView(xin, &xv);
329:   for (i = 0; i < ngroup; i++) { /* 8 y's per group */
330:     for (j = 0; j < 8; j++) VecGetKokkosView(yin[cur + j], &yv[j]);
331:     MDotFunctor mdot(xv, 8, yv[0], yv[1], yv[2], yv[3], yv[4], yv[5], yv[6], yv[7]); /* Hope Kokkos make it asynchronous */
332:     Kokkos::parallel_reduce(Kokkos::RangePolicy<WorkTag>(0, N), mdot, Kokkos::subview(zv, Kokkos::pair<PetscInt, PetscInt>(cur, cur + 8)));
333:     for (j = 0; j < 8; j++) VecRestoreKokkosView(yin[cur + j], &yv[j]);
334:     cur += 8;
335:   }

337:   if (rem) { /* The remaining */
338:     for (j = 0; j < rem; j++) VecGetKokkosView(yin[cur + j], &yv[j]);
339:     MDotFunctor mdot(xv, rem, yv[0], yv[1], yv[2], yv[3], yv[4], yv[5], yv[6], yv[7]);
340:     Kokkos::parallel_reduce(Kokkos::RangePolicy<WorkTag>(0, N), mdot, Kokkos::subview(zv, Kokkos::pair<PetscInt, PetscInt>(cur, cur + rem)));
341:     for (j = 0; j < rem; j++) VecRestoreKokkosView(yin[cur + j], &yv[j]);
342:   }
343:   VecRestoreKokkosView(xin, &xv);
344:   Kokkos::fence(); /* If reduce is async, then we need this fence to make sure z is ready for use on host */
345:   return 0;
346: }

409: /* y = x, where x is VECKOKKOS, but y may be not */
410: PetscErrorCode VecCopy_SeqKokkos(Vec xin, Vec yin)
411: {
412:   PetscLogGpuTimeBegin();
413:   if (xin != yin) {
414:     Vec_Kokkos *xkok = static_cast<Vec_Kokkos *>(xin->spptr);
415:     if (yin->offloadmask == PETSC_OFFLOAD_KOKKOS) {
416:       /* y is also a VecKokkos */
417:       Vec_Kokkos *ykok = static_cast<Vec_Kokkos *>(yin->spptr);
418:       /* Kokkos rule: if x's host has newer data, it will copy to y's host view; otherwise to y's device view
419:         In case x's host is newer, y's device is newer, it will error (though should not, I think). So we just
420:         clear y's sync state.
421:        */
422:       ykok->v_dual.clear_sync_state();
423:       Kokkos::deep_copy(ykok->v_dual, xkok->v_dual);
424:     } else {
425:       PetscScalar *yarray;
426:       VecGetArrayWrite(yin, &yarray);
427:       PetscScalarKokkosViewHost yv(yarray, yin->map->n);
428:       if (xkok->v_dual.need_sync_host()) { /* x's device has newer data */
429:         Kokkos::deep_copy(yv, xkok->v_dual.view_device());
430:       } else {
431:         Kokkos::deep_copy(yv, xkok->v_dual.view_host());
432:       }
433:       VecRestoreArrayWrite(yin, &yarray);
434:     }
435:   }
436:   PetscLogGpuTimeEnd();
437:   return 0;
438: }

462: /*  w = alpha x + y */
463: PetscErrorCode VecWAXPY_SeqKokkos(Vec win, PetscScalar alpha, Vec xin, Vec yin)
464: {
465:   ConstPetscScalarKokkosView xv, yv;
466:   PetscScalarKokkosView      wv;

468:   if (alpha == (PetscScalar)0.0) {
469:     VecCopy_SeqKokkos(yin, win);
470:   } else {
471:     PetscLogGpuTimeBegin();
472:     VecGetKokkosViewWrite(win, &wv);
473:     VecGetKokkosView(xin, &xv);
474:     VecGetKokkosView(yin, &yv);
475:     Kokkos::parallel_for(
476:       win->map->n, KOKKOS_LAMBDA(const int64_t i) { wv(i) = alpha * xv(i) + yv(i); });
477:     VecRestoreKokkosView(xin, &xv);
478:     VecRestoreKokkosView(yin, &yv);
479:     VecRestoreKokkosViewWrite(win, &wv);
480:     PetscLogGpuTimeEnd();
481:     PetscLogGpuFlops(2 * win->map->n);
482:   }
483:   return 0;
484: }

486: struct MAXPYFunctor {

489:   PetscScalarKokkosView      yv;

521: /*  y = y + sum alpha[i] x[i] */
522: PetscErrorCode VecMAXPY_SeqKokkos(Vec yin, PetscInt nv, const PetscScalar *alpha, Vec *xin)
523: {
524:   PetscInt                   i, j, cur = 0, ngroup = nv / 8, rem = nv % 8;
525:   PetscScalarKokkosView      yv;
526:   PetscScalar                a[8];
527:   ConstPetscScalarKokkosView xv[8];

529:   VecGetKokkosView(yin, &yv);
530:   for (i = 0; i < ngroup; i++) { /* 8 x's per group */
531:     for (j = 0; j < 8; j++) {    /* Fill the parameters */
532:       a[j] = alpha[cur + j];
533:       VecGetKokkosView(xin[cur + j], &xv[j]);
534:     }
535:     MAXPYFunctor maxpy(yv, 8, a[0], a[1], a[2], a[3], a[4], a[5], a[6], a[7], xv[0], xv[1], xv[2], xv[3], xv[4], xv[5], xv[6], xv[7]);

537:     for (j = 0; j < 8; j++) VecRestoreKokkosView(xin[cur + j], &xv[j]);
538:     cur += 8;
539:   }

541:   if (rem) { /* The remaining */
542:     for (j = 0; j < rem; j++) {
543:       a[j] = alpha[cur + j];
544:       VecGetKokkosView(xin[cur + j], &xv[j]);
545:     }
546:     MAXPYFunctor maxpy(yv, rem, a[0], a[1], a[2], a[3], a[4], a[5], a[6], a[7], xv[0], xv[1], xv[2], xv[3], xv[4], xv[5], xv[6], xv[7]);
547:     Kokkos::parallel_for(yin->map->n, maxpy);
548:     for (j = 0; j < rem; j++) VecRestoreKokkosView(xin[cur + j], &xv[j]);
549:   }
550:   VecRestoreKokkosView(yin, &yv);
551:   PetscLogGpuFlops(nv * 2.0 * yin->map->n);
552:   return 0;
553: }

717: /* A functor for DotNorm2 so that we can compute dp and nm in one kernel */
718: struct DotNorm2 {
719:   typedef PetscScalar                           value_type[];
720:   typedef ConstPetscScalarKokkosView::size_type size_type;

722:   size_type                  value_count;
723:   ConstPetscScalarKokkosView xv_, yv_; /* first and second vectors in VecDotNorm2. The order matters. */

725:   DotNorm2(ConstPetscScalarKokkosView &xv, ConstPetscScalarKokkosView &yv) : value_count(2), xv_(xv), yv_(yv) { }

727:   KOKKOS_INLINE_FUNCTION void operator()(const size_type i, value_type result) const
728:   {
729:     result[0] += PetscConj(yv_(i)) * xv_(i);
730:     result[1] += PetscConj(yv_(i)) * yv_(i);
731:   }

733:   KOKKOS_INLINE_FUNCTION void join(value_type dst, const value_type src) const
734:   {
735:     dst[0] += src[0];
736:     dst[1] += src[1];
737:   }

739:   KOKKOS_INLINE_FUNCTION void init(value_type result) const
740:   {
741:     result[0] = 0.0;
742:     result[1] = 0.0;
743:   }
744: };

991:   if (unchanged) return 0; /* If y's state has not changed since VecGetSubVector(), we only need to destroy it */

996:     VecScatterEnd(vscat, *y, x, INSERT_VALUES, SCATTER_REVERSE);
997:   } else { /* y and x's (host and device) arrays overlap */

999:     Vec_Kokkos *ykok = static_cast<Vec_Kokkos *>((*y)->spptr);

1002:     VecLockGet(x, &state);

1005:     /* The tricky part: one has to carefully sync the arrays */
1006:     if (xkok->v_dual.need_sync_device()) {      /* x's host has newer data */
1007:       ykok->v_dual.sync_host();                 /* Move y's latest values to host (since y is just a subset of x) */
1008:     } else if (xkok->v_dual.need_sync_host()) { /* x's device has newer data */
1009:       ykok->v_dual.sync_device();               /* Move y's latest data to device */
1010:     } else {                                    /* x's host and device data is already sync'ed; Copy y's sync state to x */
1011:       VecCopySyncState_Kokkos_Private(*y, x);
1012:     }
1013:     PetscObjectStateIncrease((PetscObject)x); /* Since x is updated */
1014:   }
1015:   return 0;
1016: }
