248: PETSC_INTERN PetscErrorCode DMSetUp_Stag_3d(DM dm)
249: {
250:   DM_Stag *const stag = (DM_Stag *)dm->data;
251:   PetscMPIInt    rank;
252:   PetscInt       i, j, d;
253:   PetscInt      *globalOffsets;
254:   const PetscInt dim = 3;

256:   MPI_Comm_rank(PetscObjectComm((PetscObject)dm), &rank);

258:   /* Rank grid sizes (populates stag->nRanks) */
259:   DMStagSetUpBuildRankGrid_3d(dm);

261:   /* Determine location of rank in grid */
262:   stag->rank[0] = rank % stag->nRanks[0];
263:   stag->rank[1] = rank % (stag->nRanks[0] * stag->nRanks[1]) / stag->nRanks[0];
264:   stag->rank[2] = rank / (stag->nRanks[0] * stag->nRanks[1]);
265:   for (d = 0; d < dim; ++d) {
266:     stag->firstRank[d] = PetscNot(stag->rank[d]);
267:     stag->lastRank[d]  = (PetscBool)(stag->rank[d] == stag->nRanks[d] - 1);
268:   }

270:   /* Determine locally owned region (if not already prescribed).
271:    Divide equally, giving lower ranks in each dimension and extra element if needbe.
272:    Note that this uses O(P) storage. If this ever becomes an issue, this could
273:    be refactored to not keep this data around.  */
274:   for (i = 0; i < dim; ++i) {
275:     if (!stag->l[i]) {
276:       const PetscInt Ni = stag->N[i], nRanksi = stag->nRanks[i];
277:       PetscMalloc1(stag->nRanks[i], &stag->l[i]);
278:       for (j = 0; j < stag->nRanks[i]; ++j) stag->l[i][j] = Ni / nRanksi + ((Ni % nRanksi) > j);
279:     }
280:   }

282:   /* Retrieve local size in stag->n */
283:   for (i = 0; i < dim; ++i) stag->n[i] = stag->l[i][stag->rank[i]];
284:   if (PetscDefined(USE_DEBUG)) {
285:     for (i = 0; i < dim; ++i) {
286:       PetscInt Ncheck, j;
287:       Ncheck = 0;
288:       for (j = 0; j < stag->nRanks[i]; ++j) Ncheck += stag->l[i][j];
290:     }
291:   }

293:   /* Compute starting elements */
294:   for (i = 0; i < dim; ++i) {
295:     stag->start[i] = 0;
296:     for (j = 0; j < stag->rank[i]; ++j) stag->start[i] += stag->l[i][j];
297:   }

299:   /* Determine ranks of neighbors */
300:   DMStagSetUpBuildNeighbors_3d(dm);

302:   /* Define useful sizes */
303:   {
304:     PetscInt  entriesPerEdge, entriesPerFace, entriesPerCorner, entriesPerElementRow, entriesPerElementLayer;
305:     PetscBool dummyEnd[3];
306:     for (d = 0; d < 3; ++d) dummyEnd[d] = (PetscBool)(stag->lastRank[d] && stag->boundaryType[d] != DM_BOUNDARY_PERIODIC);
307:     stag->entriesPerElement = stag->dof[0] + 3 * stag->dof[1] + 3 * stag->dof[2] + stag->dof[3];
308:     entriesPerFace          = stag->dof[0] + 2 * stag->dof[1] + stag->dof[2];
309:     entriesPerEdge          = stag->dof[0] + stag->dof[1];
310:     entriesPerCorner        = stag->dof[0];
311:     entriesPerElementRow    = stag->n[0] * stag->entriesPerElement + (dummyEnd[0] ? entriesPerFace : 0);
312:     entriesPerElementLayer  = stag->n[1] * entriesPerElementRow + (dummyEnd[1] ? stag->n[0] * entriesPerFace : 0) + (dummyEnd[1] && dummyEnd[0] ? entriesPerEdge : 0);
313:     stag->entries = stag->n[2] * entriesPerElementLayer + (dummyEnd[2] ? stag->n[0] * stag->n[1] * entriesPerFace : 0) + (dummyEnd[2] && dummyEnd[0] ? stag->n[1] * entriesPerEdge : 0) + (dummyEnd[2] && dummyEnd[1] ? stag->n[0] * entriesPerEdge : 0) + (dummyEnd[2] && dummyEnd[1] && dummyEnd[0] ? entriesPerCorner : 0);
314:   }

316:   /* Check that we will not overflow 32 bit indices (slightly overconservative) */
317:   if (!PetscDefined(USE_64BIT_INDICES)) {
319:                stag->n[0], stag->n[1], stag->n[2], stag->entriesPerElement);
320:   }

322:   /* Compute offsets for each rank into global vectors

324:     This again requires O(P) storage, which could be replaced with some global
325:     communication.
326:   */
327:   DMStagSetUpBuildGlobalOffsets_3d(dm, &globalOffsets);

329:   for (d = 0; d < dim; ++d)

332:   /* Define ghosted/local sizes */
333:   for (d = 0; d < dim; ++d) {
334:     switch (stag->boundaryType[d]) {
335:     case DM_BOUNDARY_NONE:
336:       /* Note: for a elements-only DMStag, the extra elements on the edges aren't necessary but we include them anyway */
337:       switch (stag->stencilType) {
338:       case DMSTAG_STENCIL_NONE: /* only the extra one on the right/top edges */
339:         stag->nGhost[d]     = stag->n[d];
340:         stag->startGhost[d] = stag->start[d];
341:         if (stag->lastRank[d]) stag->nGhost[d] += 1;
342:         break;
343:       case DMSTAG_STENCIL_STAR: /* allocate the corners but don't use them */
344:       case DMSTAG_STENCIL_BOX:
345:         stag->nGhost[d]     = stag->n[d];
346:         stag->startGhost[d] = stag->start[d];
347:         if (!stag->firstRank[d]) {
348:           stag->nGhost[d] += stag->stencilWidth; /* add interior ghost elements */
349:           stag->startGhost[d] -= stag->stencilWidth;
350:         }
351:         if (!stag->lastRank[d]) {
352:           stag->nGhost[d] += stag->stencilWidth; /* add interior ghost elements */
353:         } else {
354:           stag->nGhost[d] += 1; /* one element on the boundary to complete blocking */
355:         }
356:         break;
357:       default:
358:         SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unrecognized ghost stencil type %d", stag->stencilType);
359:       }
360:       break;
361:     case DM_BOUNDARY_GHOSTED:
362:       switch (stag->stencilType) {
363:       case DMSTAG_STENCIL_NONE:
364:         stag->startGhost[d] = stag->start[d];
365:         stag->nGhost[d]     = stag->n[d] + (stag->lastRank[d] ? 1 : 0);
366:         break;
367:       case DMSTAG_STENCIL_STAR:
368:       case DMSTAG_STENCIL_BOX:
369:         stag->startGhost[d] = stag->start[d] - stag->stencilWidth; /* This value may be negative */
370:         stag->nGhost[d]     = stag->n[d] + 2 * stag->stencilWidth + (stag->lastRank[d] && stag->stencilWidth == 0 ? 1 : 0);
371:         break;
372:       default:
373:         SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unrecognized ghost stencil type %d", stag->stencilType);
374:       }
375:       break;
376:     case DM_BOUNDARY_PERIODIC:
377:       switch (stag->stencilType) {
378:       case DMSTAG_STENCIL_NONE: /* only the extra one on the right/top edges */
379:         stag->nGhost[d]     = stag->n[d];
380:         stag->startGhost[d] = stag->start[d];
381:         break;
382:       case DMSTAG_STENCIL_STAR:
383:       case DMSTAG_STENCIL_BOX:
384:         stag->nGhost[d]     = stag->n[d] + 2 * stag->stencilWidth;
385:         stag->startGhost[d] = stag->start[d] - stag->stencilWidth;
386:         break;
387:       default:
388:         SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unrecognized ghost stencil type %d", stag->stencilType);
389:       }
390:       break;
391:     default:
392:       SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unsupported boundary type in dimension %" PetscInt_FMT, d);
393:     }
394:   }
395:   stag->entriesGhost = stag->nGhost[0] * stag->nGhost[1] * stag->nGhost[2] * stag->entriesPerElement;

397:   /* Create global-->local VecScatter and local->global ISLocalToGlobalMapping */
398:   DMStagSetUpBuildScatter_3d(dm, globalOffsets);
399:   DMStagSetUpBuildL2G_3d(dm, globalOffsets);

401:   /* In special cases, create a dedicated injective local-to-global map */
402:   if ((stag->boundaryType[0] == DM_BOUNDARY_PERIODIC && stag->nRanks[0] == 1) || (stag->boundaryType[1] == DM_BOUNDARY_PERIODIC && stag->nRanks[1] == 1) || (stag->boundaryType[2] == DM_BOUNDARY_PERIODIC && stag->nRanks[2] == 1)) {
403:     DMStagPopulateLocalToGlobalInjective(dm);
404:   }

406:   /* Free global offsets */
407:   PetscFree(globalOffsets);

409:   /* Precompute location offsets */
410:   DMStagComputeLocationOffsets_3d(dm);

412:   /* View from Options */
413:   DMViewFromOptions(dm, NULL, "-dm_view");

415:   return 0;
416: }

693: static PetscErrorCode DMStagSetUpBuildGlobalOffsets_3d(DM dm, PetscInt **pGlobalOffsets)
694: {
695:   const DM_Stag *const stag = (DM_Stag *)dm->data;
696:   PetscInt            *globalOffsets;
697:   PetscInt             i, j, k, d, entriesPerEdge, entriesPerFace, count;
698:   PetscMPIInt          size;
699:   PetscBool            extra[3];

701:   for (d = 0; d < 3; ++d) extra[d] = (PetscBool)(stag->boundaryType[d] != DM_BOUNDARY_PERIODIC); /* Extra points in global rep */
702:   entriesPerFace = stag->dof[0] + 2 * stag->dof[1] + stag->dof[2];
703:   entriesPerEdge = stag->dof[0] + stag->dof[1];
704:   MPI_Comm_size(PetscObjectComm((PetscObject)dm), &size);
705:   PetscMalloc1(size, pGlobalOffsets);
706:   globalOffsets    = *pGlobalOffsets;
707:   globalOffsets[0] = 0;
708:   count            = 1; /* note the count is offset by 1 here. We add the size of the previous rank */
709:   for (k = 0; k < stag->nRanks[2] - 1; ++k) {
710:     const PetscInt nnk = stag->l[2][k];
711:     for (j = 0; j < stag->nRanks[1] - 1; ++j) {
712:       const PetscInt nnj = stag->l[1][j];
713:       for (i = 0; i < stag->nRanks[0] - 1; ++i) {
714:         const PetscInt nni = stag->l[0][i];
715:         /* Interior : No right/top/front boundaries */
716:         globalOffsets[count] = globalOffsets[count - 1] + nni * nnj * nnk * stag->entriesPerElement;
717:         ++count;
718:       }
719:       {
720:         /* Right boundary - extra faces */
721:         /* i = stag->nRanks[0]-1; */
722:         const PetscInt nni   = stag->l[0][i];
723:         globalOffsets[count] = globalOffsets[count - 1] + nnj * nni * nnk * stag->entriesPerElement + (extra[0] ? nnj * nnk * entriesPerFace : 0);
724:         ++count;
725:       }
726:     }
727:     {
728:       /* j = stag->nRanks[1]-1; */
729:       const PetscInt nnj = stag->l[1][j];
730:       for (i = 0; i < stag->nRanks[0] - 1; ++i) {
731:         const PetscInt nni = stag->l[0][i];
732:         /* Up boundary - extra faces */
733:         globalOffsets[count] = globalOffsets[count - 1] + nnj * nni * nnk * stag->entriesPerElement + (extra[1] ? nni * nnk * entriesPerFace : 0);
734:         ++count;
735:       }
736:       {
737:         /* i = stag->nRanks[0]-1; */
738:         const PetscInt nni = stag->l[0][i];
739:         /* Up right boundary - 2x extra faces and extra edges */
740:         globalOffsets[count] = globalOffsets[count - 1] + nnj * nni * nnk * stag->entriesPerElement + (extra[0] ? nnj * nnk * entriesPerFace : 0) + (extra[1] ? nni * nnk * entriesPerFace : 0) + (extra[0] && extra[1] ? nnk * entriesPerEdge : 0);
741:         ++count;
742:       }
743:     }
744:   }
745:   {
746:     /* k = stag->nRanks[2]-1; */
747:     const PetscInt nnk = stag->l[2][k];
748:     for (j = 0; j < stag->nRanks[1] - 1; ++j) {
749:       const PetscInt nnj = stag->l[1][j];
750:       for (i = 0; i < stag->nRanks[0] - 1; ++i) {
751:         const PetscInt nni = stag->l[0][i];
752:         /* Front boundary - extra faces */
753:         globalOffsets[count] = globalOffsets[count - 1] + nnj * nni * nnk * stag->entriesPerElement + (extra[2] ? nni * nnj * entriesPerFace : 0);
754:         ++count;
755:       }
756:       {
757:         /* i = stag->nRanks[0]-1; */
758:         const PetscInt nni = stag->l[0][i];
759:         /* Front right boundary - 2x extra faces and extra edges */
760:         globalOffsets[count] = globalOffsets[count - 1] + nnj * nni * nnk * stag->entriesPerElement + (extra[0] ? nnk * nnj * entriesPerFace : 0) + (extra[2] ? nni * nnj * entriesPerFace : 0) + (extra[0] && extra[2] ? nnj * entriesPerEdge : 0);
761:         ++count;
762:       }
763:     }
764:     {
765:       /* j = stag->nRanks[1]-1; */
766:       const PetscInt nnj = stag->l[1][j];
767:       for (i = 0; i < stag->nRanks[0] - 1; ++i) {
768:         const PetscInt nni = stag->l[0][i];
769:         /* Front up boundary - 2x extra faces and extra edges */
770:         globalOffsets[count] = globalOffsets[count - 1] + nnj * nni * nnk * stag->entriesPerElement + (extra[1] ? nnk * nni * entriesPerFace : 0) + (extra[2] ? nnj * nni * entriesPerFace : 0) + (extra[1] && extra[2] ? nni * entriesPerEdge : 0);
771:         ++count;
772:       }
773:       /* Don't need to compute entries in last element */
774:     }
775:   }

777:   return 0;
778: }
