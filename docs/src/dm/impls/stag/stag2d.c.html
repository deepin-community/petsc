  4: /*@C
  5:   DMStagCreate2d - Create an object to manage data living on the elements, faces, and vertices of a parallelized regular 2D grid.

  7:   Collective

  9:   Input Parameters:
 10: + comm         - MPI communicator
 11: . bndx         - x boundary type, `DM_BOUNDARY_NONE`, `DM_BOUNDARY_PERIODIC`, or
 12: `DM_BOUNDARY_GHOSTED`
 13: . bndy         - y boundary type, `DM_BOUNDARY_NONE`, `DM_BOUNDARY_PERIODIC`, or `DM_BOUNDARY_GHOSTED`
 14: . M            - global number of elements in x direction
 15: . N            - global number of elements in y direction
 16: . m            - number of ranks in the x direction (may be `PETSC_DECIDE`)
 17: . n            - number of ranks in the y direction (may be `PETSC_DECIDE`)
 18: . dof0         - number of degrees of freedom per vertex/0-cell
 19: . dof1         - number of degrees of freedom per face/1-cell
 20: . dof2         - number of degrees of freedom per element/2-cell
 21: . stencilType  - ghost/halo region type: `DMSTAG_STENCIL_NONE`, `DMSTAG_STENCIL_BOX`, or `DMSTAG_STENCIL_STAR`
 22: . stencilWidth - width, in elements, of halo/ghost region
 23: . lx           - array of local x element counts, of length equal to `m`, summing to `M`
 24: - ly           - array of local y element counts, of length equal to `n`, summing to `N`

 26:   Output Parameter:
 27: . dm - the new `DMSTAG` object

 29:   Options Database Keys:
 30: + -dm_view                                      - calls `DMViewFromOptions()` at the conclusion of `DMSetUp()`
 31: . -stag_grid_x <nx>                             - number of elements in the x direction
 32: . -stag_grid_y <ny>                             - number of elements in the y direction
 33: . -stag_ranks_x <rx>                            - number of ranks in the x direction
 34: . -stag_ranks_y <ry>                            - number of ranks in the y direction
 35: . -stag_ghost_stencil_width                     - width of ghost region, in elements
 36: . -stag_boundary_type_x <none,ghosted,periodic> - `DMBoundaryType` value
 37: - -stag_boundary_type_y <none,ghosted,periodic> - `DMBoundaryType` value

 39:   Level: beginner

 41:   Notes:
 42:   You must call `DMSetUp()` after this call, before using the `DM`.
 43:   If you wish to use the options database (see the keys above) to change values in the `DMSTAG`, you must call
 44:   `DMSetFromOptions()` after this function but before `DMSetUp()`.

 46: .seealso: [](ch_stag), `DMSTAG`, `DMStagCreate1d()`, `DMStagCreate3d()`, `DMDestroy()`, `DMView()`, `DMCreateGlobalVector()`, `DMCreateLocalVector()`, `DMLocalToGlobalBegin()`, `DMDACreate2d()`
 47: @*/

 57: PETSC_INTERN PetscErrorCode DMStagRestrictSimple_2d(DM dmf, Vec xf_local, DM dmc, Vec xc_local)

 59:   PetscScalar ***LA_xf, ***LA_xc;
 60:   PetscInt       i, j, start[2], n[2], nextra[2], N[2];
 61:   PetscInt       d, dof[3];
 62:   PetscInt       slot_down_left_coarse, slot_down_left_fine;
 63:   PetscInt       slot_element_fine, slot_element_coarse;
 64:   PetscInt       slot_left_coarse, slot_down_coarse, slot_left_fine, slot_down_fine;

 66:   PetscFunctionBegin;
 67:   PetscCall(DMStagGetDOF(dmc, &dof[0], &dof[1], &dof[2], NULL));
 68:   PetscCall(DMStagGetCorners(dmc, &start[0], &start[1], NULL, &n[0], &n[1], NULL, &nextra[0], &nextra[1], NULL));
 69:   PetscCall(DMStagGetGlobalSizes(dmc, &N[0], &N[1], NULL));
 70:   if (PetscDefined(USE_DEBUG)) {
 71:     PetscInt dof_check[3], n_fine[2], start_fine[2];

 73:     PetscCall(DMStagGetDOF(dmf, &dof_check[0], &dof_check[1], &dof_check[2], NULL));
 74:     PetscCall(DMStagGetCorners(dmf, &start_fine[0], &start_fine[1], NULL, &n_fine[0], &n_fine[1], NULL, NULL, NULL, NULL));
 75:     for (d = 0; d < 3; ++d) PetscCheck(dof_check[d] == dof[d], PetscObjectComm((PetscObject)dmf), PETSC_ERR_ARG_INCOMP, "Cannot transfer between DMStag objects with different dof on each stratum");
 76:     for (d = 0; d < 2; ++d) PetscCheck(n_fine[d] == 2 * n[d], PetscObjectComm((PetscObject)dmf), PETSC_ERR_ARG_INCOMP, "Cannot transfer between DMStag objects unless there is a 2-1 coarsening");
 77:     for (d = 0; d < 2; ++d) PetscCheck(start_fine[d] == 2 * start[d], PetscObjectComm((PetscObject)dmf), PETSC_ERR_ARG_INCOMP, "Cannot transfer between DMStag objects unless there is a 2-1 coarsening");
 78:     {
 79:       PetscInt size_local, entries_local;

 81:       PetscCall(DMStagGetEntriesLocal(dmf, &entries_local));
 82:       PetscCall(VecGetLocalSize(xf_local, &size_local));
 83:       PetscCheck(entries_local == size_local, PETSC_COMM_WORLD, PETSC_ERR_ARG_INCOMP, "Fine vector must be a local vector of size %" PetscInt_FMT ", but a vector of size %" PetscInt_FMT " was supplied", entries_local, size_local);
 84:     }
 85:     {
 86:       PetscInt size_local, entries_local;

 88:       PetscCall(DMStagGetEntriesLocal(dmc, &entries_local));
 89:       PetscCall(VecGetLocalSize(xc_local, &size_local));
 90:       PetscCheck(entries_local == size_local, PETSC_COMM_WORLD, PETSC_ERR_ARG_INCOMP, "Coarse vector must be a local vector of size %" PetscInt_FMT ", but a vector of size %" PetscInt_FMT " was supplied", entries_local, size_local);
 91:     }
 92:   }
 93:   PetscCall(VecZeroEntries(xc_local));
 94:   PetscCall(DMStagVecGetArray(dmf, xf_local, &LA_xf));
 95:   PetscCall(DMStagVecGetArray(dmc, xc_local, &LA_xc));
 96:   PetscCall(DMStagGetLocationSlot(dmf, DMSTAG_DOWN_LEFT, 0, &slot_down_left_fine));
 97:   PetscCall(DMStagGetLocationSlot(dmf, DMSTAG_LEFT, 0, &slot_left_fine));
 98:   PetscCall(DMStagGetLocationSlot(dmf, DMSTAG_DOWN, 0, &slot_down_fine));
 99:   PetscCall(DMStagGetLocationSlot(dmf, DMSTAG_ELEMENT, 0, &slot_element_fine));
100:   PetscCall(DMStagGetLocationSlot(dmc, DMSTAG_DOWN_LEFT, 0, &slot_down_left_coarse));
101:   PetscCall(DMStagGetLocationSlot(dmc, DMSTAG_LEFT, 0, &slot_left_coarse));
102:   PetscCall(DMStagGetLocationSlot(dmc, DMSTAG_DOWN, 0, &slot_down_coarse));
103:   PetscCall(DMStagGetLocationSlot(dmc, DMSTAG_ELEMENT, 0, &slot_element_coarse));
104:   for (j = start[1]; j < start[1] + n[1] + nextra[1]; j++) {
105:     for (i = start[0]; i < start[0] + n[0] + nextra[0]; i++) {
106:       for (d = 0; d < dof[0]; ++d) LA_xc[j][i][slot_down_left_coarse + d] = LA_xf[2 * j][2 * i][slot_down_left_fine + d];
107:       for (d = 0; d < dof[1]; ++d) {
108:         if (j < N[1]) LA_xc[j][i][slot_left_coarse + d] = 0.5 * (LA_xf[2 * j][2 * i][slot_left_fine + d] + LA_xf[2 * j + 1][2 * i][slot_left_fine + d]);
109:         if (i < N[0]) LA_xc[j][i][slot_down_coarse + d] = 0.5 * (LA_xf[2 * j][2 * i][slot_down_fine + d] + LA_xf[2 * j][2 * i + 1][slot_down_fine + d]);

116:     }
117:   }
118:   PetscCall(DMStagVecRestoreArray(dmf, xf_local, &LA_xf));

123: PETSC_INTERN PetscErrorCode DMStagSetUniformCoordinatesExplicit_2d(DM dm, PetscReal xmin, PetscReal xmax, PetscReal ymin, PetscReal ymax)
124: {
125:   DM_Stag       *stagCoord;
126:   DM             dmCoord;
127:   Vec            coordLocal;
128:   PetscReal      h[2], min[2];
129:   PetscScalar ***arr;
130:   PetscInt       ind[2], start_ghost[2], n_ghost[2], s, c;
131:   PetscInt       idownleft, idown, ileft, ielement;

133:   PetscFunctionBegin;
134:   PetscCall(DMGetCoordinateDM(dm, &dmCoord));
135:   stagCoord = (DM_Stag *)dmCoord->data;
136:   for (s = 0; s < 3; ++s) {
137:     PetscCheck(stagCoord->dof[s] == 0 || stagCoord->dof[s] == 2, PetscObjectComm((PetscObject)dm), PETSC_ERR_PLIB, "Coordinate DM in 2 dimensions must have 0 or 2 dof on each stratum, but stratum %" PetscInt_FMT " has %" PetscInt_FMT " dof", s,
138:                stagCoord->dof[s]);
139:   }
140:   PetscCall(DMCreateLocalVector(dmCoord, &coordLocal));

142:   PetscCall(DMStagVecGetArray(dmCoord, coordLocal, &arr));
143:   if (stagCoord->dof[0]) PetscCall(DMStagGetLocationSlot(dmCoord, DMSTAG_DOWN_LEFT, 0, &idownleft));
144:   if (stagCoord->dof[1]) {
145:     PetscCall(DMStagGetLocationSlot(dmCoord, DMSTAG_DOWN, 0, &idown));
146:     PetscCall(DMStagGetLocationSlot(dmCoord, DMSTAG_LEFT, 0, &ileft));
147:   }
148:   if (stagCoord->dof[2]) PetscCall(DMStagGetLocationSlot(dmCoord, DMSTAG_ELEMENT, 0, &ielement));
149:   PetscCall(DMStagGetGhostCorners(dmCoord, &start_ghost[0], &start_ghost[1], NULL, &n_ghost[0], &n_ghost[1], NULL));

151:   min[0] = xmin;
152:   min[1] = ymin;
153:   h[0]   = (xmax - xmin) / stagCoord->N[0];
154:   h[1]   = (ymax - ymin) / stagCoord->N[1];

156:   for (ind[1] = start_ghost[1]; ind[1] < start_ghost[1] + n_ghost[1]; ++ind[1]) {
157:     for (ind[0] = start_ghost[0]; ind[0] < start_ghost[0] + n_ghost[0]; ++ind[0]) {
158:       if (stagCoord->dof[0]) {
159:         const PetscReal offs[2] = {0.0, 0.0};
160:         for (c = 0; c < 2; ++c) arr[ind[1]][ind[0]][idownleft + c] = min[c] + ((PetscReal)ind[c] + offs[c]) * h[c];
161:       }
162:       if (stagCoord->dof[1]) {
163:         const PetscReal offs[2] = {0.5, 0.0};
164:         for (c = 0; c < 2; ++c) arr[ind[1]][ind[0]][idown + c] = min[c] + ((PetscReal)ind[c] + offs[c]) * h[c];
165:       }
166:       if (stagCoord->dof[1]) {
167:         const PetscReal offs[2] = {0.0, 0.5};
168:         for (c = 0; c < 2; ++c) arr[ind[1]][ind[0]][ileft + c] = min[c] + ((PetscReal)ind[c] + offs[c]) * h[c];
169:       }
170:       if (stagCoord->dof[2]) {
171:         const PetscReal offs[2] = {0.5, 0.5};
172:         for (c = 0; c < 2; ++c) arr[ind[1]][ind[0]][ielement + c] = min[c] + ((PetscReal)ind[c] + offs[c]) * h[c];
173:       }
174:     }
175:   }
176:   PetscCall(DMStagVecRestoreArray(dmCoord, coordLocal, &arr));
177:   PetscCall(DMSetCoordinatesLocal(dm, coordLocal));
178:   PetscCall(VecDestroy(&coordLocal));
179:   PetscFunctionReturn(PETSC_SUCCESS);
180: }

184: static PetscErrorCode DMStagSetUpBuildNeighbors_2d(DM);
185: static PetscErrorCode DMStagSetUpBuildGlobalOffsets_2d(DM, PetscInt **);

188: PETSC_INTERN PetscErrorCode DMSetUp_Stag_2d(DM dm)
189: {
190:   DM_Stag *const stag = (DM_Stag *)dm->data;
191:   PetscMPIInt    size, rank;
192:   PetscInt       i, j, d, entriesPerElementRowGhost, entriesPerCorner, entriesPerFace, entriesPerElementRow;
193:   MPI_Comm       comm;
194:   PetscInt      *globalOffsets;
195:   PetscBool      star, dummyStart[2], dummyEnd[2];
196:   const PetscInt dim = 2;

198:   PetscFunctionBegin;
199:   PetscCall(PetscObjectGetComm((PetscObject)dm, &comm));
200:   PetscCallMPI(MPI_Comm_size(comm, &size));
201:   PetscCallMPI(MPI_Comm_rank(comm, &rank));

203:   /* Rank grid sizes (populates stag->nRanks) */
204:   PetscCall(DMStagSetUpBuildRankGrid_2d(dm));

206:   /* Determine location of rank in grid (these get extra boundary points on the last element)
207:      Order is x-fast, as usual */
208:   stag->rank[0] = rank % stag->nRanks[0];
209:   stag->rank[1] = rank / stag->nRanks[0];
210:   for (i = 0; i < dim; ++i) {
211:     stag->firstRank[i] = PetscNot(stag->rank[i]);
212:     stag->lastRank[i]  = (PetscBool)(stag->rank[i] == stag->nRanks[i] - 1);
213:   }

215:   /* Determine Locally owned region

217:    Divide equally, giving lower ranks in each dimension and extra element if needbe.

219:    Note that this uses O(P) storage. If this ever becomes an issue, this could
220:    be refactored to not keep this data around.  */
221:   for (i = 0; i < dim; ++i) {
222:     if (!stag->l[i]) {
223:       const PetscInt Ni = stag->N[i], nRanksi = stag->nRanks[i];
224:       PetscCall(PetscMalloc1(stag->nRanks[i], &stag->l[i]));
225:       for (j = 0; j < stag->nRanks[i]; ++j) stag->l[i][j] = Ni / nRanksi + ((Ni % nRanksi) > j);
226:     }
227:   }

229:   /* Retrieve local size in stag->n */
230:   for (i = 0; i < dim; ++i) stag->n[i] = stag->l[i][stag->rank[i]];
231:   if (PetscDefined(USE_DEBUG)) {
232:     for (i = 0; i < dim; ++i) {
233:       PetscInt Ncheck, j;
234:       Ncheck = 0;
235:       for (j = 0; j < stag->nRanks[i]; ++j) Ncheck += stag->l[i][j];
236:       PetscCheck(Ncheck == stag->N[i], PETSC_COMM_SELF, PETSC_ERR_ARG_SIZ, "Local sizes in dimension %" PetscInt_FMT " don't add up. %" PetscInt_FMT " != %" PetscInt_FMT, i, Ncheck, stag->N[i]);
237:     }
238:   }

240:   /* Compute starting elements */
241:   for (i = 0; i < dim; ++i) {
242:     stag->start[i] = 0;
243:     for (j = 0; j < stag->rank[i]; ++j) stag->start[i] += stag->l[i][j];
244:   }

246:   /* Determine ranks of neighbors, using DMDA's convention

248:      n6 n7 n8
249:      n3    n5
250:      n0 n1 n2                                               */
251:   PetscCall(DMStagSetUpBuildNeighbors_2d(dm));

253:   /* Determine whether the ghost region includes dummies or not. This is currently
254:        equivalent to having a non-periodic boundary. If not, then
255:        ghostOffset{Start,End}[d] elements correspond to elements on the neighbor.
256:        If true, then
257:        - at the start, there are ghostOffsetStart[d] ghost elements
258:        - at the end, there is a layer of extra "physical" points inside a layer of
259:          ghostOffsetEnd[d] ghost elements
260:        Note that this computation should be updated if any boundary types besides
261:        NONE, GHOSTED, and PERIODIC are supported.  */
262:   for (d = 0; d < 2; ++d) dummyStart[d] = (PetscBool)(stag->firstRank[d] && stag->boundaryType[d] != DM_BOUNDARY_PERIODIC);
263:   for (d = 0; d < 2; ++d) dummyEnd[d] = (PetscBool)(stag->lastRank[d] && stag->boundaryType[d] != DM_BOUNDARY_PERIODIC);

265:   /* Define useful sizes */
266:   stag->entriesPerElement = stag->dof[0] + 2 * stag->dof[1] + stag->dof[2];
267:   entriesPerFace          = stag->dof[0] + stag->dof[1];
268:   entriesPerCorner        = stag->dof[0];
269:   entriesPerElementRow    = stag->n[0] * stag->entriesPerElement + (dummyEnd[0] ? entriesPerFace : 0);
270:   stag->entries           = stag->n[1] * entriesPerElementRow + (dummyEnd[1] ? stag->n[0] * entriesPerFace : 0) + (dummyEnd[0] && dummyEnd[1] ? entriesPerCorner : 0);

272:   /* Compute offsets for each rank into global vectors
273:      This again requires O(P) storage, which could be replaced with some global
274:      communication.  */
275:   PetscCall(DMStagSetUpBuildGlobalOffsets_2d(dm, &globalOffsets));

277:   for (d = 0; d < dim; ++d)
278:     PetscCheck(stag->boundaryType[d] == DM_BOUNDARY_NONE || stag->boundaryType[d] == DM_BOUNDARY_PERIODIC || stag->boundaryType[d] == DM_BOUNDARY_GHOSTED, PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unsupported boundary type");

280:   /* Define ghosted/local sizes */
281:   if (stag->stencilType != DMSTAG_STENCIL_NONE && (stag->n[0] < stag->stencilWidth || stag->n[1] < stag->stencilWidth)) {
282:     SETERRQ(PETSC_COMM_SELF, PETSC_ERR_SUP, "DMStag 2d setup does not support local sizes (%" PetscInt_FMT " x %" PetscInt_FMT ") smaller than the elementwise stencil width (%" PetscInt_FMT ")", stag->n[0], stag->n[1], stag->stencilWidth);
283:   }
284:   for (d = 0; d < dim; ++d) {
285:     switch (stag->boundaryType[d]) {
286:     case DM_BOUNDARY_NONE:
287:       /* Note: for a elements-only DMStag, the extra elements on the faces aren't necessary but we include them anyway */
288:       switch (stag->stencilType) {
289:       case DMSTAG_STENCIL_NONE: /* only the extra one on the right/top faces */
290:         stag->nGhost[d]     = stag->n[d];
291:         stag->startGhost[d] = stag->start[d];
292:         if (stag->lastRank[d]) stag->nGhost[d] += 1;
293:         break;
294:       case DMSTAG_STENCIL_STAR: /* allocate the corners but don't use them */
295:       case DMSTAG_STENCIL_BOX:
296:         stag->nGhost[d]     = stag->n[d];
297:         stag->startGhost[d] = stag->start[d];
298:         if (!stag->firstRank[d]) {
299:           stag->nGhost[d] += stag->stencilWidth; /* add interior ghost elements */
300:           stag->startGhost[d] -= stag->stencilWidth;
301:         }
302:         if (!stag->lastRank[d]) {
303:           stag->nGhost[d] += stag->stencilWidth; /* add interior ghost elements */
304:         } else {
305:           stag->nGhost[d] += 1; /* one element on the boundary to complete blocking */
306:         }
307:         break;
308:       default:
309:         SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unrecognized ghost stencil type %d", stag->stencilType);
310:       }
311:       break;
312:     case DM_BOUNDARY_GHOSTED:
313:       switch (stag->stencilType) {
314:       case DMSTAG_STENCIL_NONE:
315:         stag->startGhost[d] = stag->start[d];
316:         stag->nGhost[d]     = stag->n[d] + (stag->lastRank[d] ? 1 : 0);
317:         break;
318:       case DMSTAG_STENCIL_STAR:
319:       case DMSTAG_STENCIL_BOX:
320:         stag->startGhost[d] = stag->start[d] - stag->stencilWidth; /* This value may be negative */
321:         stag->nGhost[d]     = stag->n[d] + 2 * stag->stencilWidth + (stag->lastRank[d] && stag->stencilWidth == 0 ? 1 : 0);
322:         break;
323:       default:
324:         SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unrecognized ghost stencil type %d", stag->stencilType);
325:       }
326:       break;
327:     case DM_BOUNDARY_PERIODIC:
328:       switch (stag->stencilType) {
329:       case DMSTAG_STENCIL_NONE: /* only the extra one on the right/top faces */
330:         stag->nGhost[d]     = stag->n[d];
331:         stag->startGhost[d] = stag->start[d];
332:         break;
333:       case DMSTAG_STENCIL_STAR:
334:       case DMSTAG_STENCIL_BOX:
335:         stag->nGhost[d]     = stag->n[d] + 2 * stag->stencilWidth;
336:         stag->startGhost[d] = stag->start[d] - stag->stencilWidth;
337:         break;
338:       default:
339:         SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unrecognized ghost stencil type %d", stag->stencilType);
340:       }
341:       break;
342:     default:
343:       SETERRQ(PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Unsupported boundary type in dimension %" PetscInt_FMT, d);
344:     }
345:   }
346:   stag->entriesGhost        = stag->nGhost[0] * stag->nGhost[1] * stag->entriesPerElement;
347:   entriesPerElementRowGhost = stag->nGhost[0] * stag->entriesPerElement;

349:   /* Create global-->local VecScatter and local->global ISLocalToGlobalMapping

351:      We iterate over all local points twice. First, we iterate over each neighbor, populating
352:      1. idxLocal[] : the subset of points, in local numbering ("S" from 0 on all points including ghosts), which correspond to global points. That is, the set of all non-dummy points in the ghosted representation
353:      2. idxGlobal[]: the corresponding global points, in global numbering (Nested "S"s - ranks then non-ghost points in each rank)

355:      Next, we iterate over all points in the local ordering, populating
356:      3. idxGlobalAll[] : entry i is the global point corresponding to local point i, or -1 if local point i is a dummy.

358:      Note further here that the local/ghosted vectors:
359:      - Are always an integral number of elements-worth of points, in all directions.
360:      - Contain three flavors of points:
361:      1. Points which "live here" in the global representation
362:      2. Ghost points which correspond to points on other ranks in the global representation
363:      3. Ghost points, which we call "dummy points," which do not correspond to any point in the global representation

365:      Dummy ghost points arise in at least three ways:
366:      1. As padding for the right, top, and front physical boundaries, to complete partial elements
367:      2. As unused space in the "corners" on interior ranks when using a star stencil
368:      3. As additional work space on all physical boundaries, when DM_BOUNDARY_GHOSTED is used

370:      Note that, because of the boundary dummies,
371:      with a stencil width of zero, on 1 rank, local and global vectors
372:      are still different!

986: static PetscErrorCode DMStagSetUpBuildRankGrid_2d(DM dm)
987: {
988:   DM_Stag *const stag = (DM_Stag *)dm->data;
989:   PetscInt       m, n;
990:   PetscMPIInt    rank, size;
991:   const PetscInt M = stag->N[0];
992:   const PetscInt N = stag->N[1];

994:   PetscFunctionBegin;
995:   PetscCallMPI(MPI_Comm_size(PetscObjectComm((PetscObject)dm), &size));
996:   PetscCallMPI(MPI_Comm_rank(PetscObjectComm((PetscObject)dm), &rank));
997:   m = stag->nRanks[0];
998:   n = stag->nRanks[1];
999:   if (m != PETSC_DECIDE) {
1000:     PetscCheck(m >= 1, PetscObjectComm((PetscObject)dm), PETSC_ERR_ARG_OUTOFRANGE, "Non-positive number of ranks in X direction: %" PetscInt_FMT, m);
1001:     PetscCheck(m <= size, PetscObjectComm((PetscObject)dm), PETSC_ERR_ARG_OUTOFRANGE, "Too many ranks in X direction: %" PetscInt_FMT " %d", m, size);
1002:   }
1003:   if (n != PETSC_DECIDE) {
1004:     PetscCheck(n >= 1, PetscObjectComm((PetscObject)dm), PETSC_ERR_ARG_OUTOFRANGE, "Non-positive number of ranks in Y direction: %" PetscInt_FMT, n);
1005:     PetscCheck(n <= size, PetscObjectComm((PetscObject)dm), PETSC_ERR_ARG_OUTOFRANGE, "Too many ranks in Y direction: %" PetscInt_FMT " %d", n, size);
1006:   }
1007:   if (m == PETSC_DECIDE || n == PETSC_DECIDE) {
1008:     if (n != PETSC_DECIDE) {
1009:       m = size / n;
1010:     } else if (m != PETSC_DECIDE) {
1011:       n = size / m;
1012:     } else {
1013:       /* try for squarish distribution */
1014:       m = (PetscInt)(0.5 + PetscSqrtReal(((PetscReal)M) * ((PetscReal)size) / ((PetscReal)N)));
1015:       if (!m) m = 1;
1016:       while (m > 0) {
1017:         n = size / m;
1018:         if (m * n == size) break;
1019:         m--;
1020:       }
1021:       if (M > N && m < n) {
1022:         PetscInt _m = m;
1023:         m           = n;
1024:         n           = _m;
1025:       }
1026:     }
1027:     PetscCheck(m * n == size, PetscObjectComm((PetscObject)dm), PETSC_ERR_PLIB, "Unable to create partition, check the size of the communicator and input m and n ");
1028:   } else PetscCheck(m * n == size, PetscObjectComm((PetscObject)dm), PETSC_ERR_ARG_OUTOFRANGE, "Given Bad partition. Product of sizes (%" PetscInt_FMT ") does not equal communicator size (%d)", m * n, size);
1029:   PetscCheck(M >= m, PetscObjectComm((PetscObject)dm), PETSC_ERR_ARG_OUTOFRANGE, "Partition in x direction is too fine! %" PetscInt_FMT " %" PetscInt_FMT, M, m);
1030:   PetscCheck(N >= n, PetscObjectComm((PetscObject)dm), PETSC_ERR_ARG_OUTOFRANGE, "Partition in y direction is too fine! %" PetscInt_FMT " %" PetscInt_FMT, N, n);
1031:   stag->nRanks[0] = m;
1032:   stag->nRanks[1] = n;
1033:   PetscFunctionReturn(PETSC_SUCCESS);
1034: }

1036: static PetscErrorCode DMStagSetUpBuildNeighbors_2d(DM dm)
1037: {
1038:   DM_Stag *const stag = (DM_Stag *)dm->data;
1039:   PetscInt       d, i;
1040:   PetscBool      per[2], first[2], last[2];
1041:   PetscInt       neighborRank[9][2], r[2], n[2];
1042:   const PetscInt dim = 2;

1044:   PetscFunctionBegin;
1045:   for (d = 0; d < dim; ++d)
1046:     PetscCheck(stag->boundaryType[d] == DM_BOUNDARY_NONE || stag->boundaryType[d] == DM_BOUNDARY_PERIODIC || stag->boundaryType[d] == DM_BOUNDARY_GHOSTED, PetscObjectComm((PetscObject)dm), PETSC_ERR_SUP, "Neighbor determination not implemented for %s",
1047:                DMBoundaryTypes[stag->boundaryType[d]]);

1049:   /* Assemble some convenience variables */
1050:   for (d = 0; d < dim; ++d) {
1051:     per[d]   = (PetscBool)(stag->boundaryType[d] == DM_BOUNDARY_PERIODIC);
1052:     first[d] = stag->firstRank[d];
1053:     last[d]  = stag->lastRank[d];
1054:     r[d]     = stag->rank[d];
1055:     n[d]     = stag->nRanks[d];
1056:   }

1058:   /* First, compute the position in the rank grid for all neighbors */
1059:   neighborRank[0][0] = first[0] ? (per[0] ? n[0] - 1 : -1) : r[0] - 1; /* left  down */
1060:   neighborRank[0][1] = first[1] ? (per[1] ? n[1] - 1 : -1) : r[1] - 1;

1062:   neighborRank[1][0] = r[0]; /*       down */
1063:   neighborRank[1][1] = first[1] ? (per[1] ? n[1] - 1 : -1) : r[1] - 1;

1065:   neighborRank[2][0] = last[0] ? (per[0] ? 0 : -1) : r[0] + 1; /* right down */
1066:   neighborRank[2][1] = first[1] ? (per[1] ? n[1] - 1 : -1) : r[1] - 1;

1068:   neighborRank[3][0] = first[0] ? (per[0] ? n[0] - 1 : -1) : r[0] - 1; /* left       */
1069:   neighborRank[3][1] = r[1];

1071:   neighborRank[4][0] = r[0]; /*            */
1072:   neighborRank[4][1] = r[1];

1074:   neighborRank[5][0] = last[0] ? (per[0] ? 0 : -1) : r[0] + 1; /* right      */
1075:   neighborRank[5][1] = r[1];

1077:   neighborRank[6][0] = first[0] ? (per[0] ? n[0] - 1 : -1) : r[0] - 1; /* left  up   */
1078:   neighborRank[6][1] = last[1] ? (per[1] ? 0 : -1) : r[1] + 1;

1080:   neighborRank[7][0] = r[0]; /*       up   */
1081:   neighborRank[7][1] = last[1] ? (per[1] ? 0 : -1) : r[1] + 1;

1083:   neighborRank[8][0] = last[0] ? (per[0] ? 0 : -1) : r[0] + 1; /* right up   */
1084:   neighborRank[8][1] = last[1] ? (per[1] ? 0 : -1) : r[1] + 1;

1086:   /* Then, compute the rank of each in the linear ordering */
1087:   PetscCall(PetscMalloc1(9, &stag->neighbors));
1088:   for (i = 0; i < 9; ++i) {
1089:     if (neighborRank[i][0] >= 0 && neighborRank[i][1] >= 0) {
1090:       stag->neighbors[i] = neighborRank[i][0] + n[0] * neighborRank[i][1];
1091:     } else {
1092:       stag->neighbors[i] = -1;
1093:     }
1094:   }

1096:   PetscFunctionReturn(PETSC_SUCCESS);
1097: }

1135:       ++count;
1136:     }
1137:     /* Don't need to compute entries in last element */
1138:   }
1139:   PetscFunctionReturn(PETSC_SUCCESS);
1140: }

1162: PETSC_INTERN PetscErrorCode DMStagPopulateLocalToGlobalInjective_2d(DM dm)
1163: {
1164:   DM_Stag *const  stag = (DM_Stag *)dm->data;
1165:   PetscInt       *idxLocal, *idxGlobal, *globalOffsetsRecomputed;
1166:   const PetscInt *globalOffsets;
1167:   PetscInt        i, j, d, count, entriesPerCorner, entriesPerFace, entriesPerElementRowGhost, entriesPerElementRow, ghostOffsetStart[2];
1168:   IS              isLocal, isGlobal;
1169:   PetscBool       dummyEnd[2];

1171:   PetscFunctionBegin;
1172:   PetscCall(DMStagSetUpBuildGlobalOffsets_2d(dm, &globalOffsetsRecomputed)); /* note that we don't actually use all of these. An available optimization is to pass them, when available */
1173:   globalOffsets = globalOffsetsRecomputed;
1174:   PetscCall(PetscMalloc1(stag->entries, &idxLocal));
1175:   PetscCall(PetscMalloc1(stag->entries, &idxGlobal));
1176:   for (d = 0; d < 2; ++d) dummyEnd[d] = (PetscBool)(stag->lastRank[d] && stag->boundaryType[d] != DM_BOUNDARY_PERIODIC);
1177:   entriesPerCorner          = stag->dof[0];
1178:   entriesPerFace            = stag->dof[0] + stag->dof[1];
1179:   entriesPerElementRow      = stag->n[0] * stag->entriesPerElement + (dummyEnd[0] ? entriesPerFace : 0);
1180:   entriesPerElementRowGhost = stag->nGhost[0] * stag->entriesPerElement;
1181:   count                     = 0;
1182:   for (d = 0; d < 2; ++d) ghostOffsetStart[d] = stag->start[d] - stag->startGhost[d];
1183:   {
1184:     const PetscInt neighbor     = 4;
1185:     const PetscInt globalOffset = globalOffsets[stag->neighbors[neighbor]];
1186:     for (j = 0; j < stag->n[1]; ++j) {
1187:       const PetscInt jghost = j + ghostOffsetStart[1];
1188:       for (i = 0; i < stag->n[0]; ++i) {
1189:         const PetscInt ighost = i + ghostOffsetStart[0];
1190:         for (d = 0; d < stag->entriesPerElement; ++d, ++count) {
1191:           idxGlobal[count] = globalOffset + j * entriesPerElementRow + i * stag->entriesPerElement + d;
1192:           idxLocal[count]  = jghost * entriesPerElementRowGhost + ighost * stag->entriesPerElement + d;
1193:         }
1194:       }
1195:       if (dummyEnd[0]) {
1196:         const PetscInt ighost = i + ghostOffsetStart[0];
1197:         i                     = stag->n[0];
1198:         for (d = 0; d < stag->dof[0]; ++d, ++count) { /* vertex first */
1199:           idxGlobal[count] = globalOffset + j * entriesPerElementRow + i * stag->entriesPerElement + d;
1200:           idxLocal[count]  = jghost * entriesPerElementRowGhost + ighost * stag->entriesPerElement + d;
1201:         }
1202:         for (d = 0; d < stag->dof[1]; ++d, ++count) { /* then left edge (skipping bottom face) */
1203:           idxGlobal[count] = globalOffset + j * entriesPerElementRow + i * stag->entriesPerElement + stag->dof[0] + d;
1204:           idxLocal[count]  = jghost * entriesPerElementRowGhost + ighost * stag->entriesPerElement + stag->dof[0] + stag->dof[1] + d;
1205:         }
1206:       }
1207:     }
1208:     if (dummyEnd[1]) {
1209:       const PetscInt jghost = j + ghostOffsetStart[1];
1210:       j                     = stag->n[1];
1211:       for (i = 0; i < stag->n[0]; ++i) {
1212:         const PetscInt ighost = i + ghostOffsetStart[0];
1213:         for (d = 0; d < entriesPerFace; ++d, ++count) {                                        /* vertex and bottom face (which are the first entries) */
1214:           idxGlobal[count] = globalOffset + j * entriesPerElementRow + i * entriesPerFace + d; /* note i increment by entriesPerFace */
1215:           idxLocal[count]  = jghost * entriesPerElementRowGhost + ighost * stag->entriesPerElement + d;
1216:         }
1217:       }
1218:       if (dummyEnd[0]) {
1219:         const PetscInt ighost = i + ghostOffsetStart[0];
1220:         i                     = stag->n[0];
1221:         for (d = 0; d < entriesPerCorner; ++d, ++count) {                                      /* vertex only */
1222:           idxGlobal[count] = globalOffset + j * entriesPerElementRow + i * entriesPerFace + d; /* note i increment by entriesPerFace */
1223:           idxLocal[count]  = jghost * entriesPerElementRowGhost + ighost * stag->entriesPerElement + d;
1224:         }
1225:       }
1226:     }
1227:   }
1228:   PetscCall(ISCreateGeneral(PetscObjectComm((PetscObject)dm), stag->entries, idxLocal, PETSC_OWN_POINTER, &isLocal));
1229:   PetscCall(ISCreateGeneral(PetscObjectComm((PetscObject)dm), stag->entries, idxGlobal, PETSC_OWN_POINTER, &isGlobal));
1230:   {
1231:     Vec local, global;
1232:     PetscCall(VecCreateMPIWithArray(PetscObjectComm((PetscObject)dm), 1, stag->entries, PETSC_DECIDE, NULL, &global));
1233:     PetscCall(VecCreateSeqWithArray(PETSC_COMM_SELF, stag->entriesPerElement, stag->entriesGhost, NULL, &local));
1234:     PetscCall(VecScatterCreate(local, isLocal, global, isGlobal, &stag->ltog_injective));
1235:     PetscCall(VecDestroy(&global));
1236:     PetscCall(VecDestroy(&local));
1237:   }
1238:   PetscCall(ISDestroy(&isLocal));
1239:   PetscCall(ISDestroy(&isGlobal));
1240:   if (globalOffsetsRecomputed) PetscCall(PetscFree(globalOffsetsRecomputed));
1241:   PetscFunctionReturn(PETSC_SUCCESS);
1242: }
