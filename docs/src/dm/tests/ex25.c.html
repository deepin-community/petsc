  1: static char help[] = "Tests DMLocalToGlobal() for dof > 1\n\n";

  3: #include <petscdm.h>
  4: #include <petscdmda.h>

  6: int main(int argc, char **argv)
  7: {
  8:   PetscInt        M = 6, N = 5, P = 4, m = PETSC_DECIDE, n = PETSC_DECIDE, p = PETSC_DECIDE, i, j, k, is, js, ks, in, jen, kn;
  9:   DM              da;
 10:   Vec             local, global;
 11:   PetscScalar ****l;

 13:   PetscFunctionBeginUser;
 14:   PetscCall(PetscInitialize(&argc, &argv, (char *)0, help));
 15:   /* Create distributed array and get vectors */
 16:   PetscCall(DMDACreate3d(PETSC_COMM_WORLD, DM_BOUNDARY_NONE, DM_BOUNDARY_NONE, DM_BOUNDARY_NONE, DMDA_STENCIL_BOX, M, N, P, m, n, p, 2, 1, NULL, NULL, NULL, &da));
 17:   PetscCall(DMSetFromOptions(da));
 18:   PetscCall(DMSetUp(da));
 19:   PetscCall(DMCreateGlobalVector(da, &global));
 20:   PetscCall(DMCreateLocalVector(da, &local));

 22:   PetscCall(DMDAGetCorners(da, &is, &js, &ks, &in, &jen, &kn));
 23:   PetscCall(DMDAVecGetArrayDOF(da, local, &l));
 24:   for (i = is; i < is + in; i++) {
 25:     for (j = js; j < js + jen; j++) {
 26:       for (k = ks; k < ks + kn; k++) {
 27:         l[k][j][i][0] = 2 * (i + j * M + k * M * N);
 28:         l[k][j][i][1] = 2 * (i + j * M + k * M * N) + 1;
 29:       }
 30:     }
 31:   }
 32:   PetscCall(DMDAVecRestoreArrayDOF(da, local, &l));
 33:   PetscCall(DMLocalToGlobalBegin(da, local, ADD_VALUES, global));
 34:   PetscCall(DMLocalToGlobalEnd(da, local, ADD_VALUES, global));

 36:   PetscCall(VecView(global, PETSC_VIEWER_STDOUT_WORLD));

 38:   /* Free memory */
 39:   PetscCall(VecDestroy(&local));
 40:   PetscCall(VecDestroy(&global));
 41:   PetscCall(DMDestroy(&da));
 42:   PetscCall(PetscFinalize());
 43:   return 0;
 44: }
