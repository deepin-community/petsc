 22: /*@C
 23:   PetscViewerRegisterAll - Registers all of the viewer methods in the `PetscViewer` package.

 25:   Not Collective

 27:    Level: developer

 29: .seealso: [](sec_viewers), `PetscViewer`
 30: @*/
 31: PetscErrorCode PetscViewerRegisterAll(void)
 32: {
 33:   if (PetscViewerRegisterAllCalled) return 0;
 34:   PetscViewerRegisterAllCalled = PETSC_TRUE;

 36:   PetscViewerRegister(PETSCVIEWERASCII, PetscViewerCreate_ASCII);
 37:   PetscViewerRegister(PETSCVIEWERBINARY, PetscViewerCreate_Binary);
 38:   PetscViewerRegister(PETSCVIEWERSTRING, PetscViewerCreate_String);
 39:   PetscViewerRegister(PETSCVIEWERDRAW, PetscViewerCreate_Draw);
 40: #if defined(PETSC_USE_SOCKET_VIEWER)
 41:   PetscViewerRegister(PETSCVIEWERSOCKET, PetscViewerCreate_Socket);
 42: #endif
 43: #if defined(PETSC_HAVE_MATHEMATICA)
 44:   PetscViewerRegister(PETSCVIEWERMATHEMATICA, PetscViewerCreate_Mathematica);
 45: #endif
 46:   PetscViewerRegister(PETSCVIEWERVU, PetscViewerCreate_VU);
 47: #if defined(PETSC_HAVE_HDF5)
 48:   PetscViewerRegister(PETSCVIEWERHDF5, PetscViewerCreate_HDF5);
 49: #endif
 50: #if defined(PETSC_HAVE_MATLAB)
 51:   PetscViewerRegister(PETSCVIEWERMATLAB, PetscViewerCreate_Matlab);
 52: #endif
 53: #if defined(PETSC_HAVE_SAWS)
 54:   PetscViewerRegister(PETSCVIEWERSAWS, PetscViewerCreate_SAWs);
 55: #endif
 56:   PetscViewerRegister(PETSCVIEWERVTK, PetscViewerCreate_VTK);
 57:   PetscViewerRegister(PETSCVIEWERGLVIS, PetscViewerCreate_GLVis);
 58: #if defined(PETSC_HAVE_ADIOS)
 59:   PetscViewerRegister(PETSCVIEWERADIOS, PetscViewerCreate_ADIOS);
 60: #endif
 61: #if defined(PETSC_HAVE_EXODUSII)
 62:   PetscViewerRegister(PETSCVIEWEREXODUSII, PetscViewerCreate_ExodusII);
 63: #endif
 64: #if defined(PETSC_HAVE_CGNS)
 65:   PetscViewerRegister(PETSCVIEWERCGNS, PetscViewerCreate_CGNS);
 66: #endif
 67:   return 0;
 68: }
